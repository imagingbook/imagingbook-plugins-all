009package Fourier_Shape_Descriptors;
010
011import ij.IJ;
012import ij.ImagePlus;
013import ij.gui.Overlay;
014import ij.gui.Roi;
015import ij.gui.ShapeRoi;
016import ij.plugin.filter.PlugInFilter;
017import ij.process.ByteProcessor;
018import ij.process.ImageProcessor;
019import imagingbook.lib.math.Complex;
020import imagingbook.pub.fd.FourierDescriptor;
021import imagingbook.pub.fd.FourierDescriptorUniform;
022import imagingbook.pub.regions.Contour;
023import imagingbook.pub.regions.RegionContourLabeling;
024
025import java.awt.geom.Point2D;
026import java.util.List;
027
028
029/**
030 * This plugin corresponds to the Fourier descriptor code example in the book's implementation 
031 * section (Program 6.1 in UtICS Vol. 3).
032 * The input image ip is assumed to contain a binary image. The class 'RegionContourLabeling'
033 * is used to find connected regions. Then the list of outer contours is retrieved and the 
034 * longest contour is assigned to 'V' as an array of type 'Point2D'. Then the contour 'V' is 
035 * used to create a Fourier descriptor with 15 coefficient pairs. Alternatively, we could have 
036 * created a Fourier descriptor of the same length (number of coefficients) as the contour and 
037 * then truncated it (using the'truncate()' method) to the specified number of coefficient 
038 * pairs. A partial reconstruction of the contour (with 100 sample points) is calculated 
039 * from the Fourier descriptor 'fd'. Finally, a pair of invariant descriptors (contained in the 
040 * array 'fdAB') is calculated.
041 * 
042 * Note: This plugin only works with BYTE (grayscale) images, pixel values &gt; 0 are assumed to be 
043 * foreground. Lookup-tables are ignored.
044 * 
045 * @author W. Burger
046 * @version 2013/05/30
047 */
048
049public class Fourier_Descriptor_RegularSampling extends CommonSetup implements PlugInFilter {
050                
051        private ImagePlus img;
052
053        public int setup(String arg, ImagePlus img) { 
054        this.img = img;
055                return DOES_8G + NO_CHANGES; 
056        }
057        
058        public void run(ImageProcessor ip) {
059                if (!setParameters()) 
060                        return;
061                
062                ByteProcessor bp = (ByteProcessor) ip;
063                
064                // segment the image and select the longest outer region contour:
065                RegionContourLabeling labeling = new RegionContourLabeling(bp);
066                List<Contour> outerContours = labeling.getOuterContours(true);
067                if (outerContours.isEmpty()) {
068                        IJ.error("no regions found");
069                        return;
070                }
071                Contour contr = outerContours.get(0);   // select the longest contour
072                Point2D[] V = contr.getPointArray();
073                
074                // create the Fourier descriptor for 'V' with Mp coefficient pairs:
075                int Mp = FourierDescriptorPairs;
076                FourierDescriptor fd = new FourierDescriptorUniform(V, Mp);
077                
078                // reconstruct the corresponding shape with 100 contour points:
079                Complex[] R = fd.getReconstruction(100);
080                
081                // create a pair of invariant descriptors (G^A, G^B):
082                FourierDescriptor[] fdAB = fd.makeInvariant();
083                FourierDescriptor fdA = fdAB[0];        // = G^A
084                FourierDescriptor fdB = fdAB[1];        // = G^B
085
086                // ----------------------------------------------------------------
087                // show various reconstructions (as a vector overlay)
088                // ----------------------------------------------------------------
089                
090                Overlay oly = new Overlay();
091                
092                if (DrawOriginalContour) {
093                        Roi roi = makeClosedPathShape(fd.getSamples(), 0.5, 0.5);
094                        roi.setStrokeColor(ContourColor);
095                        roi.setStrokeWidth(ContourStrokeWidth);
096                        oly.add(roi);
097                }
098                
099                if (DrawOriginalSamplePoints) {
100                        for (Complex c : fd.getSamples()) {
101                                ShapeRoi roi = makeCircleShape(c, SampleRadius, 0.5, 0.5);
102                                roi.setStrokeColor(SampleStrokeColor);
103                                roi.setStrokeWidth(SampleStrokeWidth);
104                                if (SampleFillColor != null) {
105                                        roi.setFillColor(SampleFillColor);
106                                }
107                                oly.add(roi);
108                        }
109                }
110                
111                if (DrawShapeCenter) {
112                        ShapeRoi roi = makeCrossShape(fd.getCoefficient(0), 2, 0.5, 0.5);
113                        roi.setStrokeColor(ReconstructionColor);
114                        roi.setStrokeWidth(ReconstructionStrokeWidth);
115                        oly.add(roi);
116                }
117        
118                if (DrawReconstruction) {
119                        ShapeRoi roi = makeClosedPathShape(R, 0.5, 0.5);
120                        roi.setStrokeColor(ReconstructionColor);
121                        roi.setStrokeWidth(ReconstructionStrokeWidth);
122                        oly.add(roi);
123                }
124                
125                if (DrawNormalizedShapes) {
126                        ShapeRoi roiA = makeClosedPathShape(fdA.getReconstruction(ShapeReconstructionPoints), 0.5, 0.5);
127                        roiA.setStrokeColor(CanonicalShapeAColor);
128                        roiA.setStrokeWidth(ReconstructionStrokeWidth);
129                        oly.add(roiA);
130                        ShapeRoi roiB = makeClosedPathShape(fdB.getReconstruction(ShapeReconstructionPoints), 0.5, 0.5);
131                        roiB.setStrokeColor(CanonicalShapeBColor);
132                        roiB.setStrokeWidth(ReconstructionStrokeWidth);
133                        oly.add(roiB);
134                }
135                
136                String title = img.getShortTitle() + "-Fourier-Descriptors";
137                ByteProcessor ip2 = (ByteProcessor) ip.duplicate();
138                ImagePlus im = new ImagePlus(title, ip2);       
139                if (ip2.isInvertedLut()) {
140                        ip2.invert();
141                        ip2.invertLut();
142                }
143                brighten(ip2, 220);
144                im.setOverlay(oly);
145                im.show();
146        }
147        
148}
