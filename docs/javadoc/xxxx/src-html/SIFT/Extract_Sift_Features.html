010package SIFT;
011
012import ij.IJ;
013import ij.ImagePlus;
014import ij.gui.GenericDialog;
015import ij.gui.Overlay;
016import ij.gui.Roi;
017import ij.gui.ShapeRoi;
018import ij.plugin.filter.PlugInFilter;
019import ij.process.FloatProcessor;
020import ij.process.ImageProcessor;
021import imagingbook.lib.ij.IjLogStream;
022import imagingbook.pub.sift.SiftDescriptor;
023import imagingbook.pub.sift.SiftDetector;
024import imagingbook.pub.sift.util.Colors;
025
026import java.awt.Color;
027import java.awt.geom.Path2D;
028import java.util.Collections;
029import java.util.List;
030
031/**
032 * This plugin extracts multi-scale SIFT features from the current 
033 * image and displays them as M-shaped markers.
034 * List of keypoints (if selected) is sorted by descending magnitude. 
035 * @author W. Burger
036 * @version 2015/05/22
037 */
038
039public class Extract_Sift_Features implements PlugInFilter {
040        
041        static {
042                IjLogStream.redirectSystem();
043        }
044
045        static double FeatureScale = 1.0; // 1.5;
046        static double FeatureStrokewidth = 1.0;
047        static double DisplayAngleOffset = -Math.PI / 2;
048        
049        static boolean ListSiftFeatures = false;
050
051        ImagePlus imp;
052        Color[] colors = Colors.defaultDisplayColors;
053
054        public int setup(String arg0, ImagePlus imp) {
055                this.imp = imp;
056                return DOES_8G + NO_CHANGES;
057        }
058
059        public void run(ImageProcessor ip) {
060                FloatProcessor fp = (FloatProcessor) ip.convertToFloat();
061                SiftDetector.Parameters params = new SiftDetector.Parameters();
062                
063                if (!setParameters(params)) {
064                        return;
065                }
066                
067                SiftDetector sd = new SiftDetector(fp, params);
068                
069                IJ.log("Extracting SIFT features ...");
070                List<SiftDescriptor> features = sd.getSiftFeatures();
071                IJ.log("SIFT features found: " + features.size());
072                
073                if (ListSiftFeatures) {
074                        int i = 0;
075                        for (SiftDescriptor d : features) {
076                                IJ.log(i + ": " + d.toString());
077                                i++;
078                        }
079                }
080
081                ImageProcessor ip2 = ip.duplicate();
082                ImagePlus imp2 = new ImagePlus(imp.getShortTitle() + "-SIFT", ip2);
083        
084                Overlay oly = drawToOverlay(features);  // show key points (with orientation)
085
086                if (oly != null) {
087                        imp2.setOverlay(oly);
088                        imp2.show();
089                }
090        }
091        
092        boolean setParameters(SiftDetector.Parameters params) {
093                        GenericDialog gd = new GenericDialog("Set SIFT parameters");
094                        gd.addNumericField("tMag :", params.t_Mag, 3, 6, "");
095                        gd.addNumericField("rMax :", params.rho_Max, 3, 6, "");
096                        gd.addNumericField("orientation histogram smoothing :", params.n_Smooth, 0, 6, "");
097                        gd.addCheckbox("list all SIFT features (might be many!)", ListSiftFeatures);
098                        gd.showDialog();
099                        if (gd.wasCanceled()) {
100                                return false;
101                        }
102                        params.t_Mag = gd.getNextNumber();
103                        params.rho_Max = gd.getNextNumber();
104                        params.n_Smooth = (int) gd.getNextNumber();
105                        ListSiftFeatures = gd.getNextBoolean();
106                        return true;
107        }
108
109        Overlay drawToOverlay(List<SiftDescriptor> desc) {
110                Collections.reverse(desc);
111                Overlay oly = new Overlay();
112                for (SiftDescriptor sd : desc) {
113                        Roi shapeRoi = makeFeatureShape(sd);
114                        shapeRoi.setStrokeWidth((float) FeatureStrokewidth);
115                        shapeRoi.setStrokeColor(colors[0]);
116                        oly.add(shapeRoi);
117                }
118                return oly;
119        }
120        
121        private ShapeRoi makeFeatureShape(SiftDescriptor sd) {
122                final double x = sd.getX(); 
123                final double y = sd.getY(); 
124                final double scale = FeatureScale * sd.getScale();
125                final double orient = sd.getOrientation() + DisplayAngleOffset;
126                final double sin = Math.sin(orient);
127                final double cos = Math.cos(orient);
128                Path2D poly = new Path2D.Double();      
129                poly.moveTo(x + (sin - cos) * scale, y - (sin + cos) * scale);
130                poly.lineTo(x + (sin + cos) * scale, y + (sin - cos) * scale);
131                poly.lineTo(x, y);
132                poly.lineTo(x - (sin - cos) * scale, y + (sin + cos) * scale);
133                poly.lineTo(x - (sin + cos) * scale, y - (sin - cos) * scale);
134                poly.closePath();
135                return new ShapeRoi(poly);
136        }
137        
138}
