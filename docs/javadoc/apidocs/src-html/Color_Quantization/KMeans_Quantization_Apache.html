010package Color_Quantization;
011
012import ij.ImagePlus;
013import ij.gui.GenericDialog;
014import ij.plugin.filter.PlugInFilter;
015import ij.process.ByteProcessor;
016import ij.process.ColorProcessor;
017import ij.process.ImageProcessor;
018import imagingbook.lib.ij.IjLogStream;
019import imagingbook.pub.color.quantize.ColorQuantizer;
020import imagingbook.pub.color.quantize.KMeansClusteringQuantizerApache;
021import imagingbook.pub.color.quantize.KMeansClusteringQuantizerApache.Parameters;
022
023
024/**
025 * ImageJ plugin demonstrating the use of the {@link KMeansClusteringQuantizerApache} class.
026 * 
027 * @author WB
028 * @version 2017/01/03
029 */
030public class KMeans_Quantization_Apache implements PlugInFilter {
031        
032        private static boolean CREATE_INDEXED_IMAGE = true; 
033        private static boolean CREATE_RGB_IMAGE = false;
034        private static boolean LIST_COLOR_TABLE = false;
035        
036        static {
037                IjLogStream.redirectSystem();
038        }
039        
040        public int setup(String arg, ImagePlus imp) {
041                return DOES_RGB + NO_CHANGES;
042        }
043        
044        public void run(ImageProcessor ip) {
045                Parameters params = new Parameters();
046        
047                if (!showDialog(params))
048                        return;
049                
050                ColorProcessor cp = (ColorProcessor) ip;
051                int[] pixels = (int[]) cp.getPixels();
052                
053                // create a quantizer object
054                ColorQuantizer quantizer = new KMeansClusteringQuantizerApache(pixels, params);
055                int nCols = quantizer.getColorMap().length;
056                
057                if (CREATE_INDEXED_IMAGE) {
058                        // quantize to an indexed color image
059                        ByteProcessor idxIp = quantizer.quantize(cp);
060                        (new ImagePlus("Quantized Index Color Image (" + nCols + " colors)", idxIp)).show();
061                }
062                
063                if (CREATE_RGB_IMAGE) {
064                        // quantize to a full-color RGB image
065                        int[] rgbPix = quantizer.quantize(pixels);
066                        ColorProcessor rgbIp = new ColorProcessor(cp.getWidth(), cp.getHeight(), rgbPix);
067                        (new ImagePlus("Quantized RGB Image (" + nCols + " colors)" , rgbIp)).show();
068                }
069                
070                if (LIST_COLOR_TABLE) {
071                        quantizer.listColorMap();
072                }
073        }
074        
075        private boolean showDialog(Parameters params) {
076                GenericDialog gd = new GenericDialog(KMeans_Quantization_Apache.class.getSimpleName());
077                gd.addNumericField("No. of colors (2,..,256)", params.maxColors, 0);
078                gd.addNumericField("Max. iterations", params.maxIterations, 0);
079                
080                gd.addCheckbox("Create indexed color image", CREATE_INDEXED_IMAGE);
081                gd.addCheckbox("Create quantized RGB image", CREATE_RGB_IMAGE);
082                gd.addCheckbox("List quantized color table", LIST_COLOR_TABLE);
083                
084                gd.showDialog();
085                if (gd.wasCanceled())
086                        return false;
087                
088                int nc = (int) gd.getNextNumber();
089                nc = Math.min(nc, 255);
090                nc = Math.max(2, nc);
091
092                params.maxColors = nc;
093                params.maxIterations = (int) gd.getNextNumber();
094                
095                CREATE_INDEXED_IMAGE = gd.getNextBoolean();
096                CREATE_RGB_IMAGE = gd.getNextBoolean();
097                LIST_COLOR_TABLE = gd.getNextBoolean();
098                return true;
099        }
100}
