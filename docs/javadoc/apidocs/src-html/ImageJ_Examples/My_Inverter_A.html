009package ImageJ_Examples;
010
011import ij.ImagePlus;
012import ij.plugin.filter.PlugInFilter;
013import ij.process.ImageProcessor;
014
015/**
016 * This is a minimal ImageJ plugin (PlugInFilter) that inverts an
017 * 8-bit grayscale (byte) image.
018 * @author WB
019 */
020public class My_Inverter_A implements PlugInFilter {
021
022        public int setup(String args, ImagePlus im) {
023                return DOES_8G; // this plugin accepts 8-bit grayscale images 
024        }
025
026        public void run(ImageProcessor ip) {
027                int M = ip.getWidth();
028                int N = ip.getHeight();
029
030                // iterate over all image coordinates
031                for (int u = 0; u < M; u++) {
032                        for (int v = 0; v < N; v++) {
033                                int p = ip.getPixel(u, v);
034                                ip.putPixel(u, v, 255 - p);
035                        }
036                }
037        }
038
039}
