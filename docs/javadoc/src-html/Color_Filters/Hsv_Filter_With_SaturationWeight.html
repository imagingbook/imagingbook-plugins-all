009package Color_Filters;
010
011
012import ij.ImagePlus;
013import ij.plugin.filter.GaussianBlur;
014import ij.plugin.filter.PlugInFilter;
015import ij.process.ColorProcessor;
016import ij.process.FloatProcessor;
017import ij.process.ImageProcessor;
018import imagingbook.lib.math.Arithmetic;
019import imagingbook.pub.color.image.HsvConverter;
020
021/**
022 * This ImageJ plugin implements a Gaussian blur filter to the periodic
023 * hue component in HSV space.
024 * @author WB
025 *
026 */
027
028public class Hsv_Filter_With_SaturationWeight implements PlugInFilter {
029        
030        static double sigma = 3.0;
031        static boolean FILTER_ALL_COMPONENTS = true;
032        static boolean SHOW_ORIGINAL_HSV = true;
033
034        public int setup(String arg, ImagePlus imp) {
035                return DOES_RGB;
036        }
037
038        public void run(ImageProcessor ip) {
039                ColorProcessor cp = (ColorProcessor) ip;
040                final int w = ip.getWidth();
041                final int h = ip.getHeight();
042                
043                FloatProcessor fH = new FloatProcessor(w, h);
044                FloatProcessor fS = new FloatProcessor(w, h);
045                FloatProcessor fV = new FloatProcessor(w, h);
046                
047                if (SHOW_ORIGINAL_HSV) {
048                        new ImagePlus("H", fH).show();
049                        new ImagePlus("S", fS).show();
050                        new ImagePlus("V", fV).show();
051                }
052                
053                // Create Cos/Sin images from the hue angle:
054                HsvConverter cc = new HsvConverter();
055                final int[] RGB = new int[3];
056                FloatProcessor fHsin = new FloatProcessor(w, h);
057                FloatProcessor fHcos = new FloatProcessor(w, h);
058                for (int v = 0; v < h; v++) {
059                        for (int u = 0; u < w; u++) {
060                                cp.getPixel(u, v, RGB);
061                                float[] HSV = cc.RGBtoHSV (RGB);        // all HSV components are in [0,1]
062                                double theta = 2 * Math.PI * HSV[0];
063                                float s = HSV[1];
064                                fHsin.setf(u, v, (float) Math.sin(theta) * s);
065                                fHcos.setf(u, v, (float) Math.cos(theta) * s);
066                                fH.setf(u, v, HSV[0]);  // hue 
067                                fS.setf(u, v, HSV[1]);  // saturation
068                                fV.setf(u, v, HSV[2]);  // value
069                        }
070                }
071
072                new ImagePlus("Hcos", fHcos).show();
073                new ImagePlus("Hsin", fHsin).show();
074                
075//              int[] hist = fH.getHistogram();
076//              new HistogramPlot(hist, "Hue histogram").show();
077                
078                
079                // Apply the Gaussian filter to Cos/Sin:
080                GaussianBlur gb = new GaussianBlur();
081                gb.blurFloat(fHcos, sigma, sigma, 0.002);
082                gb.blurFloat(fHsin, sigma, sigma, 0.002);
083                
084                if (FILTER_ALL_COMPONENTS) {
085                        gb.blurFloat(fS, sigma, sigma, 0.002);
086                        gb.blurFloat(fV, sigma, sigma, 0.002);
087                }
088                
089                // Calculate the filtered hue angle from Cos/Sin:
090                FloatProcessor fHrec = combineCosSin(fHcos, fHsin);
091                new ImagePlus("Hrec", fHrec).show();
092                // Reconstruct and show the corresponding RGB image:
093                ColorProcessor cp2 = makeRgb(fHrec, fS, fV);
094                new ImagePlus("hue blurred cos/sin", cp2).show();
095                
096//              // For testing, also show a direct blur of hue (disregarding periodicity):
097//              FloatProcessor fHblur = (FloatProcessor) fH.duplicate();
098//              gb.blurFloat(fHblur, sigma, sigma, 0.002);
099//              new ImagePlus("Hblur", fHblur).show();
100//              // Reconstruct and show the corresponding RGB image:
101//              ColorProcessor cp3 = makeRgb(fHblur, fS, fV);
102//              new ImagePlus("hue blurred direct", cp3).show();
103        }
104        
105        
106        FloatProcessor combineCosSin(FloatProcessor fHcos, FloatProcessor fHsin) {
107                final int w = fHcos.getWidth();
108                final int h = fHcos.getHeight();
109                FloatProcessor fHrec = new FloatProcessor(w, h);
110                final double twoPi = 2 * Math.PI;
111                
112                // rebuild the RGB image from blurred cos/sin
113                for (int v = 0; v < h; v++) {
114                        for (int u = 0; u < w; u++) {
115                                double Hcos = fHcos.getf(u, v);
116                                double Hsin = fHsin.getf(u, v);
117                                double theta = Math.atan2(Hsin, Hcos);    // Hpp in [-pi, .. pi]
118                                float H = (float) (Arithmetic.mod(theta, twoPi) / twoPi);  // H in [0, 1]
119                                fHrec.setf(u, v, H);
120                        }
121                }
122                return fHrec;
123        }
124        
125        /**
126         * Combine 3 FloatProcessors representing a HSV image to a RGB image.
127         * @param fH hue component (values in [0,1])
128         * @param fS saturation component (values in [0,1])
129         * @param fV value component (values in [0,1])
130         * @return A new ColorProcessor.
131         */
132        ColorProcessor makeRgb(FloatProcessor fH, FloatProcessor fS, FloatProcessor fV) {
133                final int w = fH.getWidth();
134                final int h = fH.getHeight();
135                ColorProcessor cp = new ColorProcessor(w, h);
136                HsvConverter cc = new HsvConverter();
137                
138                for (int v = 0; v < h; v++) {
139                        for (int u = 0; u < w; u++) {
140                                float H = fH.getf(u, v);
141                                float S = fS.getf(u, v);
142                                float V = fV.getf(u, v);
143                                int[] RGB2 = cc.HSVtoRGB(new float[] {H, S, V});
144                                cp.putPixel(u, v, RGB2);
145                        }
146                }
147                return cp;
148        }
149}
