009package Color_Filters;
010
011import ij.IJ;
012import ij.ImagePlus;
013import ij.gui.GenericDialog;
014import ij.plugin.filter.PlugInFilter;
015import ij.process.FloatProcessor;
016import ij.process.ImageProcessor;
017import imagingbook.lib.image.ImageMapper;
018import imagingbook.lib.interpolation.InterpolationMethod;
019import imagingbook.lib.util.Enums;
020import imagingbook.pub.color.image.ColorStack;
021import imagingbook.pub.color.image.ColorStack.ColorStackType;
022import imagingbook.pub.geometry.mappings.linear.Rotation2D;
023
024/**
025 * This plugin rotates the input image. This operation
026 * is performed in a color space specified by the user.
027 * The intent is to visualize the differences of linear interpolation
028 * when applied in different color spaces. Bilinear interpolation is 
029 * used to avoid negative component values.
030 * 
031 * @author W. Burger
032 * @version 2013/05/30
033 */
034public class Geometry_Rotate_Color implements PlugInFilter {
035        
036        static double angle = 15; // rotation angle (in degrees)
037        static ColorStackType csType = ColorStackType.sRGB;
038        
039        ImagePlus imp = null;
040        
041    public int setup(String arg, ImagePlus imp) {
042        this.imp = imp;
043        return DOES_RGB + NO_CHANGES;
044    }
045
046    public void run(ImageProcessor ip) {
047        if (!getParameters()) 
048                return;
049        ImagePlus colStack = ColorStack.createFrom(imp);
050        switch (csType) {
051                case Lab :      ColorStack.srgbToLab(colStack); break;
052                        case Luv:       ColorStack.srgbToLuv(colStack); break;
053                        case RGB:       ColorStack.srgbToRgb(colStack); break;
054                        case sRGB:      break;
055                default:
056                        IJ.error("Color space " + csType.name() + " not implemented!"); 
057                        return;
058        }
059        
060        Rotation2D imap = new Rotation2D(-2 * Math.PI * angle / 360);   // inverse mapping (target to source)
061        FloatProcessor[] processors = ColorStack.getProcessors(colStack);
062  
063        ImageMapper mapper = new ImageMapper(imap, InterpolationMethod.Bilinear);
064                for (FloatProcessor fp : processors) {
065                        //imap.applyTo(fp, InterpolationMethod.Bilinear);
066                        mapper.map(fp);
067                }
068        
069        ColorStack.toSrgb(colStack);
070        colStack.setTitle(imp.getShortTitle() + "-rotated-" + csType.name());
071        ImagePlus result = ColorStack.toColorImage(colStack);
072        result.show();
073    }
074    
075    boolean getParameters() {
076        String[] colorChoices = Enums.getEnumNames(ColorStackType.class);
077                GenericDialog gd = new GenericDialog("Gaussian Filter");
078                gd.addChoice("Color space", colorChoices, csType.name());
079        
080                gd.addNumericField("rotation angle", angle, 0);
081                gd.showDialog();
082                if(gd.wasCanceled())
083                        return false;
084                csType = ColorStackType.valueOf(gd.getNextChoice());
085                angle = gd.getNextNumber();
086
087                return true;
088    }
089
090}
