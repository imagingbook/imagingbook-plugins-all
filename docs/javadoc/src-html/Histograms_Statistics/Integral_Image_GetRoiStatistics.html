009package Histograms_Statistics;
010
011import java.awt.Rectangle;
012
013import ij.IJ;
014import ij.ImagePlus;
015import ij.gui.Roi;
016import ij.plugin.filter.PlugInFilter;
017import ij.process.ByteProcessor;
018import ij.process.ImageProcessor;
019import imagingbook.lib.image.IntegralImage;
020
021/**
022 * This ImageJ plugin calculates the integral image for the current
023 * image (8 bit grayscale only) and displays the resulting first and
024 * second order summed area tables.
025 * @author WB
026 *
027 */
028public class Integral_Image_GetRoiStatistics implements PlugInFilter {
029        
030        private ImagePlus im = null;
031
032        public int setup(String arg0, ImagePlus im) {
033                this.im = im;
034                return DOES_8G + ROI_REQUIRED + NO_CHANGES;
035        }
036
037        public void run(ImageProcessor ip) {
038                Roi roi = im.getRoi();
039                if (roi.getType() != Roi.RECTANGLE) {
040                        IJ.error("Rectangular selection required!");
041                        return;
042                }
043                
044                Rectangle rect = roi.getBounds();
045                
046                int u0 = rect.x;
047                int v0 = rect.y;
048                int u1 = u0 + rect.width - 1;
049                int v1 = v0 + rect.height - 1;
050                
051                IJ.log("rect = " + rect);
052                                
053                IntegralImage iI = new IntegralImage((ByteProcessor) ip);
054                double mean = iI.getMean(u0, v0, u1, v1);
055                double var = iI.getVariance(u0, v0, u1, v1);
056                
057                IJ.log("ROI area = " + rect.width * rect.height);
058                IJ.log("ROI mean = " + String.format("%.3f", mean));
059                IJ.log("ROI variance = " + String.format("%.3f", var));
060                IJ.log("ROI stddev = " + String.format("%.3f", Math.sqrt(var)));
061
062        }
063        
064        float[][] toFloatArray(long[][] A) {
065                float[][] B = new float[A.length][A[0].length];
066                for (int i = 0; i < A.length; i++) {
067                        for (int j = 0; j < A[0].length; j++) {
068                                B[i][j] = (float) A[i][j];
069                        }
070                }
071                return B;
072        }
073
074}
