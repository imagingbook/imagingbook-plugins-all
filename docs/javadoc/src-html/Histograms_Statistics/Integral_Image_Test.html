009package Histograms_Statistics;
010
011import ij.ImagePlus;
012import ij.plugin.filter.PlugInFilter;
013import ij.process.ByteProcessor;
014import ij.process.FloatProcessor;
015import ij.process.ImageProcessor;
016import imagingbook.lib.image.IntegralImage;
017
018/**
019 * This ImageJ plugin calculates the integral image for the current
020 * image (8 bit grayscale only) and displays the resulting first and
021 * second order summed area tables.
022 * @author WB
023 *
024 */
025public class Integral_Image_Test implements PlugInFilter {
026
027        public int setup(String arg0, ImagePlus im) {
028                return DOES_8G + NO_CHANGES;
029        }
030
031        public void run(ImageProcessor ip) {
032                        
033                IntegralImage iI = new IntegralImage((ByteProcessor) ip);
034                
035                long[][] S1 = iI.getS1();
036                long[][] S2 = iI.getS2();
037                
038                FloatProcessor fS1 = new FloatProcessor(toFloatArray(S1));
039                new ImagePlus("S1", fS1).show();
040                
041                FloatProcessor fS2 = new FloatProcessor(toFloatArray(S2));
042                new ImagePlus("S2", fS2).show();
043        }
044        
045        float[][] toFloatArray(long[][] A) {
046                float[][] B = new float[A.length][A[0].length];
047                for (int i = 0; i < A.length; i++) {
048                        for (int j = 0; j < A[0].length; j++) {
049                                B[i][j] = (float) A[i][j];
050                        }
051                }
052                return B;
053        }
054
055}
