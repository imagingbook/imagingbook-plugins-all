009package Corner_Detection;
010
011import java.awt.Color;
012import java.util.List;
013
014import ij.IJ;
015import ij.ImagePlus;
016import ij.gui.GenericDialog;
017import ij.io.LogStream;
018import ij.plugin.filter.PlugInFilter;
019import ij.process.ColorProcessor;
020import ij.process.ImageProcessor;
021import imagingbook.pub.corners.AbstractGradientCornerDetector;
022import imagingbook.pub.corners.Corner;
023import imagingbook.pub.corners.HarrisCornerDetector;
024
025/**
026 * This plugin demonstrates the use of the Harris corner detector
027 * (see {@link HarrisCornerDetector}).
028 * It calculates the corner positions and marks them in a
029 * new color image.
030 * 
031 * @version 2020/02/25
032 */
033public class Find_Corners_Harris implements PlugInFilter {
034        
035        static {
036                LogStream.redirectSystem();
037        }
038        
039        static int nmax = 0;                                            // number of corners to show
040        static int cornerSize = 2;                                      // size of cross-markers
041        static Color cornerColor = Color.green;         // color of cross markers
042        
043        ImagePlus im;
044
045    public int setup(String arg, ImagePlus im) {
046        this.im = im;
047        return DOES_ALL + NO_CHANGES;
048    }
049    
050    public void run(ImageProcessor ip) {
051        HarrisCornerDetector.Parameters params = new HarrisCornerDetector.Parameters();
052                if (!showDialog(params)) {
053                        return;
054                }
055                
056                AbstractGradientCornerDetector cd = new HarrisCornerDetector(ip, params);
057                List<Corner> corners = cd.getCorners();
058                
059                ColorProcessor R = ip.convertToColorProcessor();
060                drawCorners(R, corners);
061                (new ImagePlus("Harris Corners from " + im.getShortTitle(), R)).show();
062    }
063    
064        private boolean showDialog(HarrisCornerDetector.Parameters params) {
065                // display dialog , return false if canceled or on error.
066                GenericDialog dlg = new GenericDialog("Harris Corner Detector");
067                dlg.addNumericField("Smoothing radius (\u03C3)", params.sigma, 3);
068                dlg.addNumericField("Sensitivity (\u03B1)", params.alpha, 3);
069                dlg.addNumericField("Corner response threshold (th)", params.tH, 0);
070                dlg.addNumericField("Border distance", params.border, 0);
071                dlg.addCheckbox("Clean up corners", params.doCleanUp);
072                dlg.addNumericField("Minimum corner distance", params.dmin, 0);
073                dlg.addNumericField("Corners to show (0 = show all)", nmax, 0);
074                dlg.showDialog();
075                if(dlg.wasCanceled())
076                        return false;
077                params.sigma = Math.max(0.5, dlg.getNextNumber());
078                params.alpha = Math.max(0, dlg.getNextNumber());
079                params.tH = dlg.getNextNumber();
080                params.border = (int) dlg.getNextNumber();
081                params.doCleanUp = dlg.getNextBoolean();
082                params.dmin = (int) dlg.getNextNumber();
083                nmax = (int) dlg.getNextNumber();
084                if(dlg.invalidNumber()) {
085                        IJ.error("Input Error", "Invalid input number");
086                        return false;
087                }       
088                return true;
089        }
090        
091        //-------------------------------------------------------------------
092        
093        private void drawCorners(ImageProcessor ip, List<Corner> corners) {
094                ip.setColor(cornerColor);
095                int n = 0;
096                for (Corner c : corners) {
097                        c.draw(ip, cornerSize);
098                        n = n + 1;
099                        if (nmax > 0 && n >= nmax)
100                                break;
101                }
102        }
103        
104        // Brightens the image ip. May not work with ShortProcessor and FloatProcessor
105        @SuppressWarnings("unused")
106        private void brighten(ImageProcessor ip) {      
107                int[] lookupTable = new int[256];
108                for (int i = 0; i < 256; i++) {
109                        lookupTable[i] = 128 + (i / 2);
110                }
111                ip.applyTable(lookupTable); 
112        }
113
114}
