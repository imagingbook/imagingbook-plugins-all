009package Spectral_Techniques;
010
011
012import ij.ImagePlus;
013import ij.plugin.filter.PlugInFilter;
014import ij.process.FloatProcessor;
015import ij.process.ImageProcessor;
016import imagingbook.pub.dct.Dct2d;
017
018/** 
019 * Calculates and displays the 2-dimensional DCT after converting the input image to a float image.
020 * of arbitrary size. Be patient, this is not optimized and thus slow!
021 * @author W. Burger
022 * @version 2014-04-13
023 */
024public class DCT_2D_Demo implements PlugInFilter {
025        
026        public int setup(String arg, ImagePlus im) {
027                return DOES_ALL + NO_CHANGES;
028        }
029
030        public void run(ImageProcessor ip) {
031                
032                FloatProcessor fp = ip.convertToFloatProcessor();
033                
034                // create a new DCT:
035                Dct2d dct = new Dct2d();
036                
037                // calculate the forward DCT:
038                FloatProcessor spectrum = dct.DCT(fp);
039
040                // calculate the inverse DCT:
041                FloatProcessor reconstruction = dct.iDCT(spectrum);
042                
043                // modify the spectrum for viewing and show it:
044                spectrum.abs();
045                spectrum.add(1.0);
046                spectrum.log();
047                (new ImagePlus("DCT Spectrum", spectrum)).show();
048                
049                // show the reconstructed image
050                (new ImagePlus("DCT Reconstruction", reconstruction)).show();
051        }
052        
053}
