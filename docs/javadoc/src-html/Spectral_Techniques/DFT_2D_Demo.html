009package Spectral_Techniques;
010
011import ij.ImagePlus;
012import ij.plugin.filter.PlugInFilter;
013import ij.process.FloatProcessor;
014import ij.process.ImageProcessor;
015import imagingbook.pub.dft.Dft2d;
016
017/** 
018 * Computes the 2-dimensional (power-spectrum) DFT on a float image
019 * of arbitrary size.
020 * TODO: adapt API to the DCT layout.
021 */
022public class DFT_2D_Demo implements PlugInFilter{
023
024        static boolean center = true;    //center the resulting spectrum?
025        
026        public int setup(String arg, ImagePlus imp) {
027                return DOES_ALL + NO_CHANGES;
028        }
029
030        public void run(ImageProcessor ip) {
031                FloatProcessor fp = ip.convertToFloatProcessor();
032                Dft2d dft = new Dft2d(fp, center);
033
034                ImageProcessor ipP = dft.makePowerImage();
035                ImagePlus win = new ImagePlus("DFT Power Spectrum (byte)", ipP);
036                win.show();
037        }
038
039}
