014/** 
015 * Direct implementation of the one-dimensional DFT for arbitrary signal lengths.
016 * Uses tabulated sin/cos-values for efficiency.
017
018This DFT uses the same definition as Mathematica. Example:
019        Fourier[{1, 2, 3, 4, 5, 6, 7, 8}, FourierParameters = {0, -1}]:
020        {12.7279 + 0. i, 
021        -1.41421 + 3.41421 i, 
022        -1.41421 + 1.41421 i, 
023        -1.41421 + 0.585786 i, 
024        -1.41421 + 0. i, 
025        -1.41421 - 0.585786 i, 
026        -1.41421 - 1.41421 i, 
027        -1.41421 - 3.41421 i}
028@deprecated
029*/
030public abstract class DirectDft1DTab {
031
032        //test example
033        public static void main(String[] args) {
034                double[] signal = { 1, 2, 3, 4, 5, 6, 7, 8 };
035                Complex[] g = DirectDft1D.makeComplexVector(signal);
036
037                DirectDft1D.printComplexVector(g, "Signal");
038                
039                //compute forward DFT
040                Complex[] G = DFT(g, true);
041                DirectDft1D.printComplexVector(G, "Spectrum");
042
043                //compute inverse DFT
044                Complex[] iG = DFT(G, false);
045                DirectDft1D.printComplexVector(iG, "Reconstructed signal");
046        }
047
048        //direct DFT implementation using tabulated sin/cos-values (more efficient)
049        static Complex[] DFT(Complex[] g, boolean forward) {
050                int M = g.length;
051                double[] cosTable = makeCosTable(M);
052                double[] sinTable = makeSinTable(M);
053                double s = 1 / Math.sqrt(M); //common scale factor
054                Complex[] G = new Complex[M];
055                for (int u = 0; u < M; u++) {
056                        double sumRe = 0;
057                        double sumIm = 0;
058                        for (int m = 0; m < M; m++) {
059                                double gRe = g[m].re;
060                                double gIm = g[m].im;
061                                int k = (u * m) % M;
062                                double cosPhi = cosTable[k];
063                                double sinPhi = sinTable[k];
064                                if (forward)
065                                        sinPhi = -sinPhi;
066                                //complex multiplication: (gRe + i gIm) * (cosPhi + i sinPhi)
067                                sumRe += gRe * cosPhi - gIm * sinPhi;
068                                sumIm += gRe * sinPhi + gIm * cosPhi;
069                        }
070                        G[u] = new Complex(s * sumRe, s * sumIm);
071                }
072                return G;
073        }
074        
075        static double[] makeCosTable(int M){
076                double[] table = new double[M];
077                for (int i=0; i<M; i++){
078                        table[i]= Math.cos(2*Math.PI*i/M);
079                }
080                return table;
081        }
082        
083        static double[] makeSinTable(int M){
084                double[] table = new double[M];
085                for (int i=0; i<M; i++){
086                        table[i]= Math.sin(2*Math.PI*i/M);
087                }               
088                return table;
089        }
090        
091}
