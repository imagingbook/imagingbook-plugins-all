009package Spectral_Techniques.obsolete;
010
011import java.util.Locale;
012
013import imagingbook.lib.math.Complex;
014
015/** 
016 * Direct implementation of the one-dimensional DFT for arbitrary signal lengths.
017 * This DFT uses the same definition as Mathematica. Example:
018 * Fourier[{1, 2, 3, 4, 5, 6, 7, 8}, FourierParameters = {0, -1}]:
019        {12.7279 + 0. i, 
020        -1.41421 + 3.41421 i, 
021        -1.41421 + 1.41421 i, 
022        -1.41421 + 0.585786 i, 
023        -1.41421 + 0. i, 
024        -1.41421 - 0.585786 i, 
025        -1.41421 - 1.41421 i, 
026        -1.41421 - 3.41421 i}
027        @deprecated
028*/
029public abstract class DirectDft1D {
030
031        //direct DFT implementation 
032        static Complex[] DFT(Complex[] g, boolean forward) {
033                int M = g.length;
034                double s = 1 / Math.sqrt(M); //common scale factor
035                Complex[] G = new Complex[M];
036                for (int m = 0; m < M; m++) {
037                        double sumRe = 0;
038                        double sumIm = 0;
039                        double phim = 2 * Math.PI * m / M;
040                        for (int u = 0; u < M; u++) {
041                                double gRe = g[u].re;
042                                double gIm = g[u].im;
043                                double cosw = Math.cos(phim * u);
044                                double sinw = Math.sin(phim * u);
045                                if (!forward) // inverse transform
046                                        sinw = -sinw;
047                                //complex mult: (gRe + i gIm) * (cosw + i sinw)
048                                sumRe += gRe * cosw + gIm * sinw;
049                                sumIm += gIm * cosw - gRe * sinw;
050                        }
051                        G[m] = new Complex(s * sumRe, s * sumIm);
052                }
053                return G;
054        }
055        
056        public static Complex[] makeComplexVector(double[] signal) {
057                int M = signal.length;
058                Complex[] g = new Complex[M];
059                for (int i = 0; i < M; i++) {
060                        g[i] = new Complex(signal[i], 0);
061                }
062                return g;
063        }
064        
065        public static void printComplexVector(Complex[] g, String title) {
066        System.out.println("Listing of " + title);
067                for (int i = 0; i < g.length; i++) {
068                        if (g[i] == null)
069                                System.out.println(String.format("%d: ********", i));
070                        else {
071                                double gr = g[i].re;
072                                double gi = g[i].im;
073                                if (gi >= 0) {
074                                        System.out.println(String.format(Locale.US, "%d: %6.2f + %6.2fi", i, gr, Math.abs(gi)));
075                                }
076                                else {
077                                        System.out.println(String.format(Locale.US, "%d: %6.2f - %6.2fi", i, gr, Math.abs(gi)));
078                                }
079                        }
080                }
081        }
082        
083        //test example
084        public static void main(String[] args) {
085                double[] signal = { 1, 2, 3, 4, 5, 6, 7, 8 };
086                Complex[] g = makeComplexVector(signal);
087
088                printComplexVector(g, "Signal");
089                
090                //compute forward DFT
091                Complex[] G = DFT(g, true);
092                printComplexVector(G, "Spectrum");
093
094                //compute inverse DFT
095                Complex[] iG = DFT(G, false);
096                printComplexVector(iG, "Reconstructed signal");
097        }
098}
