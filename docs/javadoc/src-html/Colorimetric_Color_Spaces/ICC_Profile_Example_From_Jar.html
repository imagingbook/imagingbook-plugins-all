009package Colorimetric_Color_Spaces;
010
011import java.awt.color.ICC_ColorSpace;
012import java.awt.color.ICC_Profile;
013import java.io.IOException;
014import java.io.InputStream;
015import java.nio.file.Path;
016import java.util.ArrayList;
017import java.util.Collections;
018import java.util.List;
019
020import ij.IJ;
021import ij.gui.GenericDialog;
022import ij.plugin.PlugIn;
023import imagingbook.lib.ij.IjLogStream;
024import imagingbook.lib.math.Matrix;
025import imagingbook.lib.util.ResourceUtils;
026
027
028/**
029 * This plugin lists a user-selected ICC profile that is retrieved
030 * from a Java resource.
031 * 
032 * @author W. Burger
033 *
034 */
035public class ICC_Profile_Example_From_Jar implements PlugIn {
036        
037        static {
038                IjLogStream.redirectSystem();
039        }
040        
041        static final String ProfileDirectory = "sample-icc-profiles/";
042
043        String[] choices;
044        String theChoice;
045        
046        public void run(String arg) {
047                Class<?> clazz = imagingbook.lib.color.CssColor.class;
048                List<String> profileList = new ArrayList<String>();
049                Path[] paths = ResourceUtils.listResources(clazz, ProfileDirectory);
050                for (Path p : paths) {
051                        String name = p.getFileName().toString();
052                        profileList.add(name);
053                }
054                
055                Collections.sort(profileList);
056                choices = profileList.toArray(new String[0]);
057                
058                if (!showDialog())
059                        return;
060                
061                System.out.println("selected ICC profile: " + theChoice);
062                
063                ICC_Profile profile = null;
064                try {
065//                      InputStream strm = ResourceUtils.getResourceStream(clazz, ProfileDirectory + "/" + theChoice);
066                        InputStream strm = clazz.getResourceAsStream(ProfileDirectory + theChoice);
067                        if (strm != null)
068                                profile = ICC_Profile.getInstance(strm);
069                } catch (IOException e) { }
070                
071                System.out.println("profile = " + profile);
072
073                ICC_ColorSpace iccColorSpace = new ICC_ColorSpace(profile);
074                int nComp = iccColorSpace.getNumComponents();
075                if (nComp != 3) {
076                        IJ.error("Color space must have 3 components, this one has " + nComp);
077                        return;
078                }
079                
080                IJ.log("color space = " + iccColorSpace);
081                IJ.log("color space type = " + iccColorSpace.getType());
082                IJ.log("color space ncomp = " + iccColorSpace.getNumComponents());
083                
084                
085                // specify a device-specific color:
086                float[] deviceColor = {0.77f, 0.13f, 0.89f};
087                //float[] deviceColor = {0.0f, 0.0f, 0.0f};
088                IJ.log("device color = " + Matrix.toString(deviceColor));
089                
090                // convert to sRGB:
091                float[] sRGBColor = iccColorSpace.toRGB(deviceColor);
092                IJ.log("sRGB = " + Matrix.toString(sRGBColor));
093                
094                // convert to (D50-based) XYZ:
095                float[] XYZColor = iccColorSpace.toCIEXYZ(deviceColor);
096                IJ.log("XYZ = " + Matrix.toString(XYZColor));
097                                
098                deviceColor = iccColorSpace.fromRGB(sRGBColor);
099                IJ.log("device color direct (check) = " + Matrix.toString(deviceColor));
100                
101                deviceColor = iccColorSpace.fromCIEXYZ(XYZColor);
102                IJ.log("device color via XYZ (check) = " + Matrix.toString(deviceColor));
103                
104                // list sRGB Values:
105                for (int ri = 0; ri <= 10; ri++) {
106                        for (int gi = 0; gi <= 10; gi++) {
107                                for (int bi = 0; bi <= 10; bi++) {
108                                        float[] devCol = {ri * 0.1f, gi * 0.1f, bi * 0.1f};
109                                        float[] sRGB = iccColorSpace.toRGB(devCol);
110                                        float[] devColCheck = iccColorSpace.fromRGB(sRGB);
111                                        IJ.log(Matrix.toString(devCol) + " -> " + Matrix.toString(sRGB) + " -> " 
112                                                        + Matrix.toString(devColCheck) + warning(devCol, devColCheck));
113                                }
114                        }
115                }
116        }
117        
118        String warning(float[] col1, float[] col2) {
119                float t = 0.05f;
120                for (int i=0; i<col1.length; i++) {
121                        if (Math.abs(col1[i] - col2[i]) > t)
122                                return " ***";
123                }
124                return "";
125        }
126        
127        boolean showDialog() {
128                GenericDialog gd = new GenericDialog(ICC_Profile_Example_From_Jar.class.getSimpleName());
129                gd.addMessage("Select an ICC profile:");
130                gd.addChoice("Profile:", choices, choices[0]);
131                
132                gd.showDialog();
133                if (gd.wasCanceled())
134                        return false;
135                int mi = gd.getNextChoiceIndex();
136                theChoice = choices[mi];
137                
138                return true;
139        }
140        
141
142}
