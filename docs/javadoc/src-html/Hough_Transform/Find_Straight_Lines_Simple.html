009package Hough_Transform;
010
011import ij.IJ;
012import ij.ImagePlus;
013import ij.plugin.filter.PlugInFilter;
014import ij.process.ImageProcessor;
015import imagingbook.pub.hough.HoughTransformLines;
016import imagingbook.pub.hough.HoughTransformLines.HoughLine;
017import imagingbook.pub.hough.HoughTransformLines.Parameters;
018
019/** 
020 * This plugin implements a simple Hough transform for straight lines.
021 * It expects a binary (8-bit) image, with background = 0 and foreground (contour) 
022 * pixels with values &gt; 0.
023 * 
024 * @author WB
025 * @version 2013/08/25
026*/
027public class Find_Straight_Lines_Simple implements PlugInFilter {
028
029        public int setup(String arg, ImagePlus imp) {
030                return DOES_8G + NO_CHANGES;
031        }
032
033        public void run(ImageProcessor ip) {
034                Parameters params = new Parameters();
035                params.nAng = 256;              // = m
036                params.nRad = 256;              // = n
037
038                // compute the Hough Transform:
039                HoughTransformLines ht = new HoughTransformLines(ip, params);
040                // retrieve the 5 strongest lines with min. 50 points
041                HoughLine[] lines = ht.getLines(50, 5);
042                
043                if (lines.length > 0) {
044                        IJ.log("Lines found:");
045                        for (HoughLine L : lines) {
046                                IJ.log(L.toString());   // list the resulting lines
047                        }
048                } 
049                else 
050                        IJ.log("No lines found!");
051        }
052
053}
