009package ImageJ_Short_Reference;
010
011import ij.IJ;
012import ij.ImagePlus;
013import ij.plugin.filter.PlugInFilter;
014import ij.process.ImageProcessor;
015
016public class Display_Demo implements PlugInFilter {
017        ImagePlus im = null;
018
019        public int setup(String arg, ImagePlus imp) {
020                this.im = imp;          // keep reference to associated ImagePlus
021                return DOES_ALL;        // this plugin accepts any image
022        }
023
024        public void run(ImageProcessor ip) {
025                for (int i = 0; i < 10; i++) {
026                        // modify this image:
027                        ip.smooth();
028                        ip.rotate(30);
029                        // redisplay this image:
030                        im.updateAndDraw();
031                        // sleep so user can watch this:
032                        IJ.wait(100);
033                }
034        }
035}
