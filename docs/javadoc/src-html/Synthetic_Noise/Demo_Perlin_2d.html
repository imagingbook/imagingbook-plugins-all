009package Synthetic_Noise;
010
011import ij.ImagePlus;
012import ij.plugin.PlugIn;
013import ij.process.FloatProcessor;
014import ij.process.ImageProcessor;
015import imagingbook.pub.noise.hashing.HashFun;
016import imagingbook.pub.noise.hashing.HashPermute;
017import imagingbook.pub.noise.perlin.PerlinNoiseGen2d;
018
019/**
020 * This ImageJ plugin creates a new noise image using a 2D gradient noise 
021 * generator. Several parameters can be adjusted. For details see
022 * Ch. 8 of W. Burger and M. J. Burge. "Principles of Digital Image Processing -
023 * Advanced Methods" (Vol. 3). Undergraduate Topics in Computer Science.
024 * Springer-Verlag, London (2013). http://www.imagingbook.com
025 * 
026 * @author W. Burger
027 * @version 2013/05/28
028 */
029
030public class Demo_Perlin_2d implements PlugIn {
031        int w = 300;                            // image width
032        int h = 300;                            // image height
033        int K = 3;                                      // number of noise frequencies
034        double f_min = 0.01;            // min. frequency (in cycles per pixel unit) 
035        double f_max = f_min * Math.pow(2, K);
036        double persistence = 0.5;       // persistence
037        int seed = 2;                           // hash seed 
038        HashFun hf = new HashPermute(seed); 
039//      HashFun hf = new Hash32Ward(seed);
040//      HashFun hf = new Hash32Shift(seed);
041//      HashFun hf = new Hash32ShiftMult(seed);
042//      HashFun hf = HashFun.create(seed);
043
044        public void run(String arg0) {
045                // create the noise generator:
046                PerlinNoiseGen2d ng = new PerlinNoiseGen2d(f_min, f_max, persistence, hf);
047                // create a new image and fill with noise:
048                ImageProcessor fp = new FloatProcessor(w, h);
049                for (int v = 0; v < h; v++) {
050                        for (int u = 0; u < w; u++) {
051                                double val = ng.NOISE(u, v);
052                                fp.putPixelValue(u, v, val);
053                        }
054                }
055                // display the new image:
056                (new ImagePlus("Perlin Noise Image", fp)).show();
057        }
058}
