009package Synthetic_Noise;
010
011import imagingbook.pub.noise.hashing.HashFun;
012import imagingbook.pub.noise.perlin.PerlinNoiseGenNd;
013import ij.ImagePlus;
014import ij.plugin.PlugIn;
015import ij.process.FloatProcessor;
016import ij.process.ImageProcessor;
017
018/**
019 * This ImageJ plugin creates a new noise image using a N-dim. gradient noise 
020 * generator. Several parameters can be adjusted. For details see
021 * Ch. 8 of W. Burger and M. J. Burge. "Principles of Digital Image Processing -
022 * Advanced Methods" (Vol. 3). Undergraduate Topics in Computer Science.
023 * Springer-Verlag, London (2013). http://www.imagingbook.com
024 * 
025 * @author W. Burger
026 * @version 2013/05/28
027 */
028
029public class Demo_Perlin_Nd  implements PlugIn {
030        
031        static int w = 300;
032        static int h = 300;
033        static int octaves = 3;
034        static int seed = 2;
035
036        static double f_min = 0.01; // f_max / (1 << (octaves-1));
037        static double f_max = f_min * (1 << (octaves-1));
038        
039        static double persistence = 0.5;
040
041        String title = this.getClass().getSimpleName() + " Seed=" + seed;
042        
043        public void run(String arg0) {
044                int dim = 2;    // number of dimensions
045                
046                // choose hash function:
047                HashFun hf = HashFun.create(seed);
048//              HashFun hf = new Hash32Shift(seed);
049//              HashFun hf = new Hash32ShiftMult(seed);
050//              HashFun hf = new Hash32Ward(seed);
051//              HashFun hf = new HashPermute(seed); 
052
053                // create the noise generator:
054                PerlinNoiseGenNd png = new PerlinNoiseGenNd(dim, f_min, f_max, persistence, hf);
055                createNoiseImage(w, h, png).show();
056        }
057        
058        ImagePlus createNoiseImage(int w, int h, PerlinNoiseGenNd ng) {
059                ImageProcessor fp = new FloatProcessor(w,h);
060                double[] X = new double[2];
061                for (int v=0; v<h; v++){
062                        for (int u=0; u<w; u++){
063                                X[0] = u;
064                                X[1] = v;
065                                fp.putPixelValue(u, v, ng.NOISE(X));
066                        }
067                }
068                fp.setMinAndMax(-0.6,0.6);      //fp.resetMinAndMax();
069                ImagePlus nimg = new ImagePlus(title,fp);
070                return nimg;
071        }
072
073}
