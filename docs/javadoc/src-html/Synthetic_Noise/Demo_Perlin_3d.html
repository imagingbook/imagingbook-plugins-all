009package Synthetic_Noise;
010
011import ij.IJ;
012import ij.ImagePlus;
013import ij.ImageStack;
014import ij.gui.NewImage;
015import ij.plugin.PlugIn;
016import ij.process.ImageProcessor;
017import imagingbook.pub.noise.hashing.Hash32Shift;
018import imagingbook.pub.noise.hashing.HashFun;
019import imagingbook.pub.noise.perlin.PerlinNoiseGenNd;
020
021/**
022 * This ImageJ plugin creates a new noise image using a 3D gradient noise 
023 * generator. Several parameters can be adjusted. For details see
024 * Ch. 8 of W. Burger and M. J. Burge. "Principles of Digital Image Processing -
025 * Advanced Methods" (Vol. 3). Undergraduate Topics in Computer Science.
026 * Springer-Verlag, London (2013). http://www.imagingbook.com
027 * 
028 * @author W. Burger
029 * @version 2013/05/28
030 */
031
032public class Demo_Perlin_3d  implements PlugIn {
033        
034        static int wx = 300;
035        static int wy = 300;
036        static int wz = 10;             int stepZ = 5;
037        
038        int dim = 3;
039        
040        static int octaves = 6;
041        static int seed = 2;
042
043        static double f_min = 0.01; // f_max / (1 << (octaves-1));
044        static double f_max = f_min * (1 << (octaves-1));
045        
046        static double persistence = 0.5;
047
048        String title = this.getClass().getSimpleName() + " Seed=" + seed;
049        
050        public void run(String arg0) {
051                
052                // choose hash function:
053//              HashFun hf = HashFun.create(seed);
054                HashFun hf = new Hash32Shift(seed);
055//              HashFun hf = new Hash32ShiftMult(seed);
056//              HashFun hf = new Hash32Ward(seed);
057//              HashFun hf = new HashPermute(seed); 
058
059                // create the noise generator:
060                PerlinNoiseGenNd png = new PerlinNoiseGenNd(dim, f_min, f_max, persistence, hf);
061                createNoiseImage(wx, wy, wz, png).show();
062        }
063        
064        ImagePlus createNoiseImage(int wx, int wy, int wz, PerlinNoiseGenNd ng) {
065                ImagePlus stackImg = 
066                        NewImage.createFloatImage(title, wx, wy, wz, NewImage.FILL_BLACK);
067                ImageStack stack = stackImg.getStack();
068                double[] X = new double[dim];
069                IJ.showStatus("creating noise volume ...");
070                for (int z=1; z<=wz; z++) {
071                        IJ.showProgress(z, wz);
072                        ImageProcessor fp = stack.getProcessor(z);
073                        for (int v=0; v<wy; v++){
074                                for (int u=0; u<wx; u++){
075                                        X[0] = u;
076                                        X[1] = v;
077                                        X[2] = z * stepZ;
078                                        fp.putPixelValue(u, v, ng.NOISE(X));
079                                }
080                        }
081                        //fp.setMinAndMax(-0.6,0.6);    //fp.resetMinAndMax();
082                }
083                IJ.showStatus("");
084                stackImg.setDisplayRange(-0.6,0.6);
085                return stackImg;
086        }
087
088}
