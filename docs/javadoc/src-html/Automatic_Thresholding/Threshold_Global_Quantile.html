009package Automatic_Thresholding;
010
011import ij.IJ;
012import ij.ImagePlus;
013import ij.plugin.filter.PlugInFilter;
014import ij.process.ByteProcessor;
015import ij.process.ImageProcessor;
016import imagingbook.pub.threshold.global.GlobalThresholder;
017import imagingbook.pub.threshold.global.QuantileThresholder;
018
019/**
020 * Demo plugin showing the use of the QuantileThresholder class.
021 * @author W. Burger
022 * @version 2013/05/30
023 */
024public class Threshold_Global_Quantile implements PlugInFilter {
025        
026        static double quantile = 0.5;
027
028        public int setup(String arg, ImagePlus imp) {
029                return DOES_8G;
030        }
031        
032        public void run(ImageProcessor ip) {
033                ByteProcessor bp = (ByteProcessor) ip;
034                
035                quantile = IJ.getNumber("Black quantile [0,1]", quantile);
036                if (quantile < 0) quantile = 0;
037                if (quantile > 1) quantile = 1;
038                
039                GlobalThresholder thr = new QuantileThresholder(quantile);
040                int q = thr.getThreshold(bp);
041                if (q >= 0) {
042                        IJ.log("threshold = " + q);
043                        ip.threshold(q);
044                }
045                else {
046                        IJ.showMessage("no threshold found");
047                }
048        }
049}
