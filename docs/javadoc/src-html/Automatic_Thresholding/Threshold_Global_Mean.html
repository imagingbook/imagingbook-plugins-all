009package Automatic_Thresholding;
010
011import ij.IJ;
012import ij.ImagePlus;
013import ij.plugin.filter.PlugInFilter;
014import ij.process.ByteProcessor;
015import ij.process.ImageProcessor;
016import imagingbook.pub.threshold.global.GlobalThresholder;
017import imagingbook.pub.threshold.global.MeanThresholder;
018
019/**
020 * Demo plugin showing the use of the MeanThresholder class.
021 * @author W. Burger
022 * @version 2013/05/30
023 */
024public class Threshold_Global_Mean implements PlugInFilter {
025
026        public int setup(String arg, ImagePlus imp) {
027                return DOES_8G;
028        }
029        
030        public void run(ImageProcessor ip) {
031                ByteProcessor bp = (ByteProcessor) ip;
032
033                GlobalThresholder thr = new MeanThresholder();
034                int q = thr.getThreshold(bp);
035                if (q >= 0) {
036                        IJ.log("threshold = " + q);
037                        ip.threshold(q);
038                }
039                else {
040                        IJ.showMessage("no threshold found");
041                }       
042
043        }
044}
