009package Automatic_Thresholding;
010
011import java.util.LinkedList;
012import java.util.List;
013
014import ij.IJ;
015import ij.ImagePlus;
016import ij.gui.GenericDialog;
017import ij.plugin.filter.PlugInFilter;
018import ij.process.ImageProcessor;
019
020/**
021 * Demo plugin making available a selection of global thresholders.
022 * @author W. Burger
023 * @version 2013/05/30
024 */
025public class Threshold_Adaptive_All implements PlugInFilter {
026        
027        enum Mode {
028                Bernsen(Threshold_Adaptive_Bernsen.class), 
029                Interpolating(Threshold_Adaptive_Interpolating.class),
030//              NiblackBox(Threshold_Adaptive_NiblackBox.class),
031//              NiblackDisk(Threshold_Adaptive_NiblackDisk.class),
032//              NiblackGauss(Threshold_Adaptive_NiblackGauss.class),
033                Niblack(Threshold_Adaptive_Niblack.class),
034                Sauvola(Threshold_Adaptive_Sauvola.class);
035                
036                Class<? extends PlugInFilter> pluginClass;
037                
038                Mode(Class<? extends PlugInFilter> cls) {
039                        this.pluginClass = cls;
040                }
041                
042                static String[] getNames() {
043                        List<String> names = new LinkedList<String>();
044                        for(Mode val : Mode.values()) {
045                                names.add(val.name());
046                        }
047                        return names.toArray(new String[0]);
048                }
049        }
050        
051        ImagePlus imp = null;
052        
053        public int setup(String arg, ImagePlus imp) {
054                this.imp = imp;
055                return DOES_8G;
056        }
057
058        public void run(ImageProcessor ip) {
059                String[] choices = Mode.getNames();
060                GenericDialog gd = new GenericDialog("Global Thresholder");
061                gd.addMessage("Select an algorithm:");
062                gd.addChoice("Algorithm:", choices, choices[0]);
063                
064                gd.showDialog();
065                if (gd.wasCanceled())
066                        return;
067                
068                int mi = gd.getNextChoiceIndex();
069                Mode mode = Mode.values()[mi];
070                String pluginName = mode.pluginClass.getCanonicalName();
071                
072                imp.unlock();
073                IJ.runPlugIn(imp, pluginName, null);
074        }
075
076}
