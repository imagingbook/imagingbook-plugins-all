009package Automatic_Thresholding;
010
011import ij.IJ;
012import ij.ImagePlus;
013import ij.plugin.filter.PlugInFilter;
014import ij.process.ByteProcessor;
015import ij.process.ImageProcessor;
016
017/**
018 * Demo plugin showing MinMax thresholding.
019 * @author W. Burger
020 * @version 2013/05/30
021 */
022public class Threshold_Global_MinMax implements PlugInFilter {
023
024        private int theMin, theMax;
025        
026        public int setup(String arg, ImagePlus imp) {
027                return DOES_8G;
028        }
029        
030        public void run(ImageProcessor ip) {
031                ByteProcessor bp = (ByteProcessor) ip;
032                getMinMax(bp);
033                double minVal = theMin; 
034                double maxVal = theMax;
035                getMinMax(bp);
036                
037                int q = (int) Math.rint((minVal + maxVal)/2);
038        
039                if (q < maxVal) {
040                        IJ.log("threshold = " + q);
041                        ip.threshold(q);
042                }
043                else {
044                        IJ.showMessage("no threshold found");
045                }
046        }
047        
048        void getMinMax(ByteProcessor bp) {
049                int N = bp.getWidth() * bp.getHeight();
050                theMin = bp.get(0);
051                theMax = theMin;
052                for (int i=1; i<N; i++) {
053                        int p = bp.get(i);
054                        if (p < theMin) 
055                                theMin = p;
056                        if (p > theMax)
057                                theMax = p;
058                }
059        }
060}
