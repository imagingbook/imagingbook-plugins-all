009package _ImageJ_Examples;
010
011import ij.ImagePlus;
012import ij.plugin.filter.PlugInFilter;
013import ij.process.ImageProcessor;
014
015/**
016 * This ImageJ plugin shows how data can be communicated from one
017 * plugin to another. In this example, THIS plugin ({@link Plugin_1})
018 * calculates a histogram that is subsequently retrieved by ANOTHER
019 * plugin {@link Plugin_2}. Data are stored as a property of the associated
020 * image (of type {@link ImagePlus}).
021 * Note that the stored data should contain no instances of self-defined
022 * classes, since these may be re-loaded when performing compile-and-run.
023 * 
024 * @author W. Burger
025 *
026 */
027public class Plugin_1 implements PlugInFilter {
028        
029        public static final String HistKey = Plugin_1.class.getCanonicalName() +"histogram";
030        ImagePlus im;
031        
032        public int setup(String arg, ImagePlus im) {
033                this.im = im;
034                return DOES_ALL + NO_CHANGES;}
035                
036        public void run(ImageProcessor ip) {
037                int[] hist = ip.getHistogram();
038                // add histogram to image properties:
039                im.setProperty(HistKey, hist); 
040        }
041}
