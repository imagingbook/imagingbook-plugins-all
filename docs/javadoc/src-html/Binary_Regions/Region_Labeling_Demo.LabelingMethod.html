009package Binary_Regions;

011import ij.IJ;
012import ij.ImagePlus;
013import ij.gui.GenericDialog;
014import ij.plugin.filter.PlugInFilter;
015import ij.process.ByteProcessor;
016import ij.process.ImageProcessor;
017import imagingbook.lib.util.Enums;
018import imagingbook.pub.regions.BreadthFirstLabeling;
019import imagingbook.pub.regions.DepthFirstLabeling;
020import imagingbook.pub.regions.RecursiveLabeling;
021import imagingbook.pub.regions.RegionContourLabeling;
022import imagingbook.pub.regions.RegionLabeling;
023import imagingbook.pub.regions.RegionLabeling.BinaryRegion;
024import imagingbook.pub.regions.SequentialLabeling;
025
026import java.awt.Point;
027import java.awt.geom.Point2D;
028import java.util.List;
029
030/**
031 * This ImageJ plugin is an example for how to use the region
032 * labeling classes in the "regions" package.
033 * One of four labeling types can be selected (see run() method).
034 * They should all return the same result.
035 * 2009-11-15: Cleanup, added mu_11() example.
036 * 2010-11-19: Added user dialog, selection menu, combined region+contour 
037 * segmenter.
038*/
039public class Region_Labeling_Demo implements PlugInFilter {
040        
041        public enum LabelingMethod {
042                        BreadthFirst, DepthFirst, Recursive, Sequential, RegionAndContours
043        };
044
045        static LabelingMethod method = LabelingMethod.BreadthFirst;
046        static boolean recolor = false;
047        static boolean listRegions = true;
048        
049    public int setup(String arg, ImagePlus im) {
050                return DOES_8G + NO_CHANGES;
051    }
052        
053    public void run(ImageProcessor ip) {
054        if (!getUserInput())
055                return;
056        if (method == LabelingMethod.Recursive && 
057                        !IJ.showMessageWithCancel("Recursive labeling", "This may run out of stack memory!\n" + "Continue?")) {
058                        return;
059        }
060        
061        // Copy the original to a new byte image:
062        ByteProcessor bp = ip.convertToByteProcessor(false);
063    
064                // Select one of 4 different labeling methods:
065                RegionLabeling segmenter = null;
066                switch (method) {
067                        case BreadthFirst:              segmenter = new BreadthFirstLabeling(bp); break;
068                        case DepthFirst:                segmenter = new DepthFirstLabeling(bp); break;
069                        case Recursive:                 segmenter = new RecursiveLabeling(bp); break; 
070                        case Sequential:                segmenter = new SequentialLabeling(bp); break;
071                        case RegionAndContours: segmenter = new RegionContourLabeling(bp); break;
072                }
073
074                // Retrieve the list of detected regions:
075                List<BinaryRegion> regions = segmenter.getRegions(true);        // regions are sorted by size
076                if (listRegions) {
077                        IJ.log("Detected regions (sorted by size): " + regions.size());
078                        for (BinaryRegion r: regions) {
079                                IJ.log(r.toString());
080                        }
081                }
082                
083                // Show the resulting labeling as a random color image
084                ImageProcessor labelIp = segmenter.makeLabelImage(recolor);
085                (new ImagePlus("Label Image", labelIp)).show();
086                
087                // Example for processing all regions:
088                for (BinaryRegion r : regions) {
089                        double mu11 = mu_11(r); // example for calculating region statistics (see below)
090                        IJ.log("Region " + r.getLabel() + ": mu11=" + mu11);
091                }
092    }
093    
094        boolean getUserInput() {
095                GenericDialog gd = new GenericDialog("Binary Region Labeling");
096                String[] mNames = Enums.getEnumNames(LabelingMethod.class);
097                gd.addChoice("Labeling method", mNames, mNames[0]);
098                gd.addCheckbox("Color result", recolor);
099                gd.addCheckbox("List regions", listRegions);
100                gd.showDialog();
101                if (gd.wasCanceled()) {
102                        return false;
103                }
104                method = LabelingMethod.valueOf(gd.getNextChoice());
105                recolor = gd.getNextBoolean();
106                listRegions = gd.getNextBoolean();
107                return true;
108        }
109    
110    /*
111     * This method demonstrates how a particular region's central moment
112     * mu_11 could be calculated from the finished region labeling.
113     */
114    double mu_11 (BinaryRegion r) {
115        Point2D ctr = r.getCenterPoint();
116        final double xc = ctr.getX();   // centroid of this region
117        final double yc = ctr.getY();
118        double mu11 = 0;                                        // moment
119        
120        // iterate through all pixels of regions r:
121        for (Point p : r) {
122                mu11 = mu11 + (p.x - xc) * (p.y - yc);
123        }
124        return mu11;
125    }
126    
127}
