003import java.util.ArrayList;
004import java.util.EventListener;
005import java.util.List;
006
007import org.apache.commons.math3.linear.MatrixUtils;
008import org.apache.commons.math3.linear.RealMatrix;

013/**
014 * This class implements the Iterative Closest Point algorithm for
015 * n-dimensional samples.
016 * 
017 * @author W. Burger
018 * @deprecated
019 */
020public class IterativeClosestPointMatcher {
021        
022        /**
023         * An instance of a class implementing this interface may be passed to the constructor
024         * of {@link IterativeClosestPointMatcher}; its {@code notify()} method
025         * is invoked after every iteration of the matcher for debugging
026         * or visualizing intermediate results.
027         */
028        public interface IterationListener extends EventListener {
029                public void notify(IterativeClosestPointMatcher matcher);
030        }
031        
032        private final double tau;
033        private final int kMax;
034        private final int m, n;
035        private int[] A;
036        
037        private int k = 0;
038        private double eMin = Double.POSITIVE_INFINITY;
039        private RealMatrix T;
040        private boolean converged = false;
041        
042        private final IterationListener cb;
043        private final List<double[]> X, Y;
044        private final LeastSquaresFitter fitter;
045        
046        public IterativeClosestPointMatcher(List<double[]> X, List<double[]> Y, 
047                        double tau, int kMax, LeastSquaresFitter fitter, IterationListener cb) {
048                this.X = X;
049                this.Y = Y;
050                this.tau = tau;
051                this.kMax = kMax;
052                this.fitter = fitter;
053                this.cb = cb;
054                
055                this.m = X.size();
056                this.n = X.get(0).length;
057                this.A = new int[m];
058                
059                match();
060        }
061        
062        private void match() {
063                k = 0;
064                eMin = Double.POSITIVE_INFINITY;
065                T = initialTransformation();
066                do {
067                        System.out.println("ITERATION k = " + k);
068                        A = associatePoints();
069                        double e = fitPoints(A);        // sets T
070                        System.out.println("T = " + Matrix.toString(T.getData()));
071                        System.out.println("e = " + e);
072                        double de = eMin - e;
073                        eMin = e;
074                        k = k + 1;
075                        converged = (0 <= de && de < tau);
076                        if (cb != null) 
077                                cb.notify(this);
078                } while(!converged && k < kMax);
079        }
080
081        private RealMatrix initialTransformation() {
082                RealMatrix iT = MatrixUtils.createRealMatrix(n, n + 1);
083                iT.setSubMatrix(MatrixUtils.createRealIdentityMatrix(n).getData(), 0, 0);
084                return iT;
085        }
086        
087        private int[] associatePoints() {
088                int[] A = new int[X.size()];
089                int i = 0;
090                for (double[] xi : X) {
091                        double[] xiT = T.operate(Matrix.toHomogeneous(xi)); // T.applyTo(xi);
092                        double dMin = Double.POSITIVE_INFINITY;
093                        int jMin = 0;
094                        int j = 0;
095                        for (double[] yi : Y) {
096                                double d = Matrix.normL2squared(Matrix.subtract(xiT,yi));
097                                if (d < dMin) {
098                                        dMin = d;
099                                        jMin = j;
100                                }
101                                j = j + 1;
102                        }
103//                      System.out.format("  A[%d] = %d, %.3f\n", i, jMin, dMin);
104                        A[i] = jMin;
105                        i = i + 1;
106                }
107                return A;
108        }
109
110        private double fitPoints(int[] A) {
111                List<double[]> YY = new ArrayList<double[]>(A.length);
112
113                for (int i = 0; i < A.length; i++) {
114                        YY.add(Y.get(A[i]));
115                }
116                
117                ProcrustesFitter2 pf = new ProcrustesFitter2(X, YY, true, false, true);
118                
119                T = pf.getTransformationMatrix();
120                System.out.println("EuclideanError = " + pf.getEuclideanError(X, Y));
121                return pf.getError();
122        }
123
124        public boolean hasConverged() {
125                return converged;
126        }
127        
128        public RealMatrix getT() {
129                return T;
130        }
131
132        public int[] getA() {
133                return A;
134        }
135        
136        public int getIteration() {
137                return k;
138        }
139
140}
