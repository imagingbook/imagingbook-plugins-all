002
003import java.util.List;
004
005import org.apache.commons.math3.linear.ArrayRealVector;
006import org.apache.commons.math3.linear.LUDecomposition;
007import org.apache.commons.math3.linear.MatrixUtils;
008import org.apache.commons.math3.linear.RealMatrix;
009import org.apache.commons.math3.linear.RealVector;
010import org.apache.commons.math3.linear.SingularValueDecomposition;
011
012import imagingbook.lib.math.Matrix;
013import imagingbook.pub.geometry.mappings.linear.AffineMapping;

015/**
016 * This class implements the n-dimensional Procrustes fit algorithm described in 
017 * Shinji Umeyama, "Least-squares estimation of transformation parameters 
018 * between two point patterns", IEEE Transactions on Pattern Analysis and 
019 * Machine Intelligence 13.4 (Apr. 1991), pp. 376â€“380.
020 * The 
021 * 
022 * @author W. Burger
023 * @version 2017/03/01
024 * @deprecated
025 */
026public class ProcrustesFitter {
027        
028        final boolean allowTranslation;
029        final boolean allowScaling;
030        final boolean forceRotation;
031        
032        final int m;            // number of samples
033        final int n;            // dimension of samples
034        
035        final double c;         // scale
036        final RealMatrix R;     // orthogonal (rotation) matrix
037        final RealVector t;     // translation vector
038        final double err;       // total (squared) error
039        
040        // --------------------------------------------------------------
041        
042        /**
043         * Convenience constructor. The supplied point sequences must have the
044         * same length and order (i.e., points must be in correspondence).
045         * @param X Sequence of n-dimensional points
046         * @param Y Sequence of n-dimensional points (reference)
047         */
048        public ProcrustesFitter(List<double[]> X, List<double[]> Y) {
049                this(X, Y, true, true, true);
050        }
051        
052        /**
053         * Full constructor. The supplied point sequences must have the
054         * same length and order (i.e., points must be in correspondence).
055         * @param X Sequence of n-dimensional points
056         * @param Y Sequence of n-dimensional points (reference)
057         * @param allowTranslation If {@code true}, translation (t) between point sets is considered, 
058         *              otherwise zero translation is assumed.
059         * @param allowScaling If {@code true}, scaling (c) between point sets is considered, 
060         *              otherwise unit scale assumed.
061         * @param forceRotation If {@code true}, the orthogonal part of the transformation (Q)
062         *              is forced to a true rotation and no reflection is allowed.
063         */
064        public ProcrustesFitter(List<double[]> X, List<double[]> Y, 
065                        boolean allowTranslation, boolean allowScaling, boolean forceRotation) {
066                if (X.size() != Y.size())
067                        throw new IllegalArgumentException("point sequences xA, xB must have same length");
068                
069                this.allowTranslation = allowTranslation;
070                this.allowScaling = allowScaling;
071                this.forceRotation = forceRotation;
072                this.m = X.size();
073                this.n = X.get(0).length;
074                
075                double[] meanX = null;
076                double[] meanY = null;
077                
078                if (this.allowTranslation) {
079                        meanX = getMeanVec(X);
080                        meanY = getMeanVec(Y);
081                }
082                
083                RealMatrix P = makeDataMatrix(X, meanX);
084                RealMatrix Q = makeDataMatrix(Y, meanY);
085                MatrixUtils.checkAdditionCompatible(P, Q);      // A, B of same dimensions?
086                
087                RealMatrix QPt = Q.multiply(P.transpose());
088                SingularValueDecomposition svd = new SingularValueDecomposition(QPt);
089                
090                RealMatrix U = svd.getU();
091                RealMatrix S = svd.getS();
092                RealMatrix V = svd.getV();
093                        
094                double d = (svd.getRank() >= n) ? det(QPt) : det(U) * det(V);
095                
096                RealMatrix D = MatrixUtils.createRealIdentityMatrix(n);
097                if (d < 0 && forceRotation)
098                        D.setEntry(n - 1, n - 1, -1);
099                
100                R = U.multiply(D).multiply(V.transpose());
101                
102                double normP = P.getFrobeniusNorm();
103                double normQ = Q.getFrobeniusNorm();
104                
105                c = (this.allowScaling) ? 
106                                S.multiply(D).getTrace() / sqr(normP) : 1.0;
107                
108                if (allowTranslation) {
109                        RealVector ma = MatrixUtils.createRealVector(meanX);
110                        RealVector mb = MatrixUtils.createRealVector(meanY);
111                        t = mb.subtract(R.scalarMultiply(c).operate(ma));
112                }
113                else {
114                        t = new ArrayRealVector(n);     // zero vector
115                }
116                
117                err = sqr(normQ) - sqr(S.multiply(D).getTrace() / normP);
118        }
119        
120        // -----------------------------------------------------------------
121        
122        /**
123         * Retrieves the estimated scale.
124         * @return The estimated scale (or 1 if {@code allowscaling = false}).
125         */
126        public double getScale() {
127                return c;
128        }
129        
130        /**
131         * Retrieves the estimated orthogonal (rotation) matrix.
132         * @return The estimated rotation matrix.
133         */
134        public RealMatrix getR() {
135                return R;
136        }
137        
138        /**
139         * Retrieves the estimated translation vector.
140         * @return The estimated translation vector.
141         */
142        public RealVector getT() {
143                return t;
144        }
145        
146        /**
147         * Retrieves the total (squared) error for the estimated fit.
148         * @return The total error for the estimated fit.
149         */
150        public double getError() {
151                return err;
152        }
153        
154        /**
155         * Calculates the total error for the estimated fit as
156         * the sum of the squared Euclidean distances between the 
157         * transformed point set X and the reference set Y.
158         * This method is provided for testing as an alternative to
159         * the quicker {@link getError} method.
160         * @param X Sequence of n-dimensional points.
161         * @param Y Sequence of n-dimensional points (reference).
162         * @return The total error for the estimated fit.
163         */
164        public double getEuclideanError(List<double[]> X, List<double[]> Y) {
165                RealMatrix cR = R.scalarMultiply(c);
166                double ee = 0;
167                for (int i = 0; i < X.size(); i++) {
168                        RealVector ai = new ArrayRealVector(X.get(i));
169                        RealVector bi = new ArrayRealVector(Y.get(i));
170                        RealVector aiT = cR.operate(ai).add(t);
171                        double ei = aiT.subtract(bi).getNorm();
172                        ee = ee + sqr(ei);
173                }
174                return ee;
175        }
176        
177        /**
178         * Retrieves the (n) x (n+1) transformation matrix A, such that
179         * y_i = A * x_i (with x_i in homogeneous coordinates)
180         * @return the transformation matrix
181         */
182        public RealMatrix getTransformationMatrix() {
183                RealMatrix cR = R.scalarMultiply(c);
184                RealMatrix A = MatrixUtils.createRealMatrix(n, n + 1);
185                A.setSubMatrix(cR.getData(), 0, 0);
186                A.setColumnVector(n, t);
187                return A;
188        }
189        
190        public AffineMapping getAffineMapping2D() {
191                if (n != 2)
192                        return null;
193                AffineMapping map = new AffineMapping(
194                                c * R.getEntry(0, 0), c * R.getEntry(0, 1), t.getEntry(0),
195                                c * R.getEntry(1, 0), c * R.getEntry(1, 1), t.getEntry(1),
196                                false
197                                );
198                return map;     
199        }
200        
201        // -----------------------------------------------------------------
202        
203        private double det(RealMatrix M) {
204                return new LUDecomposition(M).getDeterminant();
205        }
206        
207        private double sqr(final double x) {
208                return x * x;
209        }
210        
211        private double[] getMeanVec(List<double[]> X) {
212                double[] sum = new double[X.get(0).length];
213                for (double[] x : X) {
214                        for (int j = 0; j < x.length; j++) {
215                                sum[j] = sum[j] + x[j];
216                        }
217                }
218                Matrix.multiplyD(1.0 / X.size(), sum);
219                return sum;
220        }
221        
222        private RealMatrix makeDataMatrix(List<double[]> X) {
223                final int m = X.size();
224                final int n = X.get(0).length;
225                RealMatrix M = MatrixUtils.createRealMatrix(n, m);
226                int i = 0;
227                for (double[] x : X) {
228                        RealVector xi = MatrixUtils.createRealVector(x);
229                        M.setColumnVector(i, xi);
230                        i++;
231                }
232                return M;
233        }
234        
235        private RealMatrix makeDataMatrix(List<double[]> X, double[] meanX) {
236                if (meanX == null) {
237                        return makeDataMatrix(X);
238                }
239                final int m = X.size();
240                final int n = X.get(0).length;
241                RealMatrix M = MatrixUtils.createRealMatrix(n, m);
242                RealVector mean = MatrixUtils.createRealVector(meanX);
243                int i = 0;
244                for (double[] x : X) {
245                        RealVector xi = MatrixUtils.createRealVector(x).subtract(mean);
246                        M.setColumnVector(i, xi);
247                        i++;
248                }
249                return M;
250        }
251        
252        void printSVD(SingularValueDecomposition svd) {
253                RealMatrix U = svd.getU();
254                RealMatrix S = svd.getS();
255                RealMatrix V = svd.getV();
256                System.out.println("------ SVD ---------------");
257                System.out.println("U = " + Matrix.toString(U.getData()));
258                System.out.println("S = " + Matrix.toString(S.getData()));
259                System.out.println("V = " + Matrix.toString(V.getData()));
260                System.out.println("--------------------------");
261        }
262
263}
