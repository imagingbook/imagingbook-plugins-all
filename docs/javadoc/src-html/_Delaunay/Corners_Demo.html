005import java.awt.geom.Path2D;

011import ij.IJ;

020import imagingbook.pub.corners.HarrisCornerDetector;
021import imagingbook.pub.geometry.basic.Point;
022import imagingbook.pub.geometry.basic.Triangle;

024import imagingbook.pub.geometry.delaunay.guibas.TriangulationGuibas;

027/**
028 * This plugin runs the Harris corner detector, applies the Delaunay
029 * triangulation to the N strongest corners and displays the result.
030 * 
031 * @author W. Burger
032 * @version 2020-01-02
033 */
034public class Corners_Demo implements PlugInFilter {
035        
036        private static int CornerCount = 0;                                     // number of corners to show (0 = show all)
037        private static Color DelaunayColor = Color.blue;        // color for graph edges
038        private static Color PointColor = Color.magenta;                // color for point markers
039        
040        private static float StrokeWidth = 0.1f;
041        private static double PointRadius = 1.5;
042        
043        private static String title = Corners_Demo.class.getSimpleName();
044        
045
046    public int setup(String arg, ImagePlus im) {
047        return DOES_ALL + NO_CHANGES;
048    }
049    
050    public void run(ImageProcessor ip) {
051                
052        HarrisCornerDetector.Parameters params = new HarrisCornerDetector.Parameters();
053                if (!showDialog(params)) {
054                        return;
055                }
056                
057                HarrisCornerDetector cd = new HarrisCornerDetector(ip, params);
058                List<Corner> corners = cd.findCorners();
059
060                DelaunayTriangulation dt = new TriangulationGuibas(toPointList(corners));
061                
062                ImageProcessor cp = ip.convertToByteProcessor();
063                Overlay oly = makeOverlay(dt);
064
065                ImagePlus im = new ImagePlus(title, cp);
066                im.setOverlay(oly);
067                im.show();
068    }
069    
070        private boolean showDialog(HarrisCornerDetector.Parameters params) {
071                // display dialog , return false if canceled or on error.
072                GenericDialog dlg = new GenericDialog("Harris Corner Detector");
073                dlg.addNumericField("Alpha", params.alpha, 3);
074                dlg.addNumericField("Threshold", params.tH, 0);
075                dlg.addCheckbox("Clean up corners", params.doCleanUp);
076                dlg.addNumericField("Corners to show (0 = show all)", CornerCount, 0);
077                dlg.showDialog();
078                if(dlg.wasCanceled())
079                        return false;   
080                params.alpha = dlg.getNextNumber();
081                params.tH = (int) dlg.getNextNumber();
082                params.doCleanUp = dlg.getNextBoolean();
083                CornerCount = (int) dlg.getNextNumber();
084                if(dlg.invalidNumber()) {
085                        IJ.error("Input Error", "Invalid input number");
086                        return false;
087                }       
088                return true;
089        }
090        
091        // ---------------------------------------------------------------------------
092        
093        /**
094         * TODO: Make class {@link Corner} implement {@link Point}.
095         * @param corners
096         * @return
097         */
098        private List<Point> toPointList(List<Corner> corners) {
099                Point[] points = new Point[corners.size()];
100                int i = 0;
101                for (Corner c : corners) {
102                        points[i] = new Point.Imp(c.getX(), c.getY());
103                        i++;
104                }
105                return Arrays.asList(points);
106        }
107        
108        private Overlay makeOverlay(DelaunayTriangulation dt) {
109                Collection<Triangle> triangles = dt.getTriangles();
110                Collection<Point> allPoints = dt.getPoints();
111                BasicStroke stroke = new BasicStroke(StrokeWidth, BasicStroke.CAP_ROUND, BasicStroke.JOIN_ROUND);
112                Overlay oly = new Overlay();
113        
114                double r = PointRadius;
115                for (Point p : allPoints) {
116                        double x = p.getX();
117                        double y = p.getY();
118                        Roi roi = new ShapeRoi(new Rectangle2D.Double(x - r, y - r, 2 * r, 2 * r));
119                        roi.setStroke(stroke);
120                        roi.setStrokeColor(PointColor);
121                        roi.setAntiAlias(true);
122                        oly.add(roi);
123                }
124
125                Path2D path = new Path2D.Double();
126                for (Triangle trgl : triangles) {
127                        Point[] pts = trgl.getPoints();
128                        Point a = pts[0];
129                        Point b = pts[1];
130                        Point c = pts[2];
131                        path.moveTo(a.getX(), a.getY());
132                        path.lineTo(b.getX(), b.getY());
133                        path.lineTo(c.getX(), c.getY());
134                        path.lineTo(a.getX(), a.getY());
135                }
136                Roi roi = new ShapeRoi(path);
137                roi.setStroke(stroke);
138                roi.setStrokeColor(DelaunayColor);
139                roi.setAntiAlias(true);
140                oly.add(roi);   
141                oly.translate(0.5, 0.5);
142                return oly;
143        }
144        
145}
