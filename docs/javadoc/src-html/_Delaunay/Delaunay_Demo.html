001package _Delaunay;
002
003import java.awt.Color;
004import java.awt.geom.Path2D;
005import java.awt.geom.Rectangle2D;
006import java.util.ArrayList;
007import java.util.Collection;
008import java.util.List;
009
010import ij.ImagePlus;
011import ij.gui.GenericDialog;
012import ij.gui.Overlay;
013import ij.gui.Roi;
014import ij.gui.ShapeRoi;
015import ij.plugin.filter.PlugInFilter;
016import ij.process.ImageProcessor;
017import imagingbook.lib.ij.IjLogStream;
018import imagingbook.pub.geometry.basic.Point;
019import imagingbook.pub.geometry.delaunay.DelaunayTriangulation;
020import imagingbook.pub.geometry.delaunay.Triangle;
021import imagingbook.pub.geometry.delaunay.chew.TriangulationChew;
022import imagingbook.pub.geometry.delaunay.guibas.TriangulationGuibas;
023
024
025/**
026 * <p>This ImageJ plugin demonstrates the calculation of of the 2D Delaunay triangulation for
027 * the given set of points.
028 * </p>
029 * <p>
030 * How to use: Create or open a binary image containing a point set. Any pixel with value greater than 0 
031 * is considered a foreground point, zero-value pixels are considered background). The plugin 
032 * triangulates the foreground point set and displays the triangulation as a vector overlay
033 * in a new image. The following triangulation methods can be selected (both should yield similar results):
034 * </p>
035 * <ol>
036 * <li><strong>Chew:</strong> Incremental triangulation algorithm proposed by Paul Chew.</li>
037 * <li><strong>Guibas:</strong> Randomized incremental algorithm proposed by Guibas et al.</li>
038 * </ol>
039 * <p>
040 * Note that the actual implementations reside in separate sub-packages and do not depend on ImageJ.
041 * </p>
042 * 
043 * @author W. Burger
044 * @version 2020-01-02
045 */
046public class Delaunay_Demo implements PlugInFilter {
047
048        private static final String title = Delaunay_Demo.class.getSimpleName();
049        
050        private static final String[] TriangulationMethods = {"Chew", "Guibas"};
051        private static int theMethod = 1;       // Guibas is 10 x faster
052        
053        private static Color DelaunayColor = Color.green;
054        private static Color PointColor = Color.magenta;
055        private static float StrokeWidth = 0.1f;
056        private static double PointRadius = 1.5;
057
058        static {
059                IjLogStream.redirectSystem();
060        }
061
062        public int setup(String arg, ImagePlus im) {
063                return DOES_ALL + NO_CHANGES;
064        }
065
066        public void run(ImageProcessor ip) {
067                if (!runDialog())
068                        return;
069
070                List<Point> points = collectPoints(ip);
071                
072                DelaunayTriangulation dt = null;
073                switch (theMethod) {
074                case 0: dt = new TriangulationChew(points); break;              // Chew
075                case 1: dt = new TriangulationGuibas(points); break;    // Guibas
076                }
077
078                ImageProcessor cp = ip.convertToByteProcessor();
079                Overlay oly = makeOverlay(dt);
080
081                ImagePlus im = new ImagePlus(title, cp);
082                im.setOverlay(oly);
083                im.show();
084        }
085
086        // ---------------------------------------------------------------------------
087
088        private boolean runDialog() {
089                GenericDialog gd = new GenericDialog(title);
090                gd.addChoice("Select triangulation method", 
091                                TriangulationMethods, TriangulationMethods[theMethod]);
092                gd.showDialog(); 
093                if (gd.wasCanceled()) 
094                        return false;
095                theMethod = gd.getNextChoiceIndex();
096                return true;
097        }
098        
099        // ---------------------------------------------------------------------------
100        
101        private List<Point> collectPoints(ImageProcessor ip) {
102                List<Point> vertices = new ArrayList<>();
103                int M = ip.getWidth();
104                int N = ip.getHeight();
105                for (int v = 0; v < N; v++) {
106                        for (int u = 0; u < M; u++) {
107                                float val = ip.getPixelValue(u, v);
108                                if (val > 0) {
109                                        vertices.add(new Point.Imp(u, v));
110                                }
111                        }
112                }
113                return vertices;
114        }
115        
116        // ---------------------------------------------------------------------------
117        
118        private Overlay makeOverlay(DelaunayTriangulation dt) {
119                Collection<Triangle> triangles = dt.getTriangles();
120                Collection<Point> allPoints = dt.getPoints();
121                Overlay oly = new Overlay();
122
123                double r = PointRadius;
124                for (Point p : allPoints) {
125                        double x = p.getX();
126                        double y = p.getY();
127                        Roi roi = new ShapeRoi(new Rectangle2D.Double(x - r, y - r, 2 * r, 2 * r));
128                        roi.setStrokeColor(PointColor);
129                        roi.setStrokeWidth(StrokeWidth);
130                        oly.add(roi);
131                }
132
133                Path2D path = new Path2D.Double();
134                for (Triangle trgl : triangles) {
135                        Point[] pts = trgl.getPoints();
136                        Point a = pts[0];
137                        Point b = pts[1];
138                        Point c = pts[2];
139                        path.moveTo(a.getX(), a.getY());
140                        path.lineTo(b.getX(), b.getY());
141                        path.lineTo(c.getX(), c.getY());
142                        path.lineTo(a.getX(), a.getY());
143                }
144                Roi roi = new ShapeRoi(path);
145                roi.setStrokeColor(DelaunayColor);
146                roi.setStrokeWidth(StrokeWidth);
147                oly.add(roi);
148                
149                oly.translate(0.5, 0.5);
150                return oly;
151        }
152}
