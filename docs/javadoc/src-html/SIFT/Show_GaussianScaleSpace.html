009package SIFT;
010
011import ij.ImagePlus;
012import ij.plugin.filter.PlugInFilter;
013import ij.process.FloatProcessor;
014import ij.process.ImageProcessor;
015import imagingbook.pub.sift.scalespace.GaussianScaleSpace;
016
017/**
018 * Builds a hierarchical Gaussian scale space representation 
019 * from the input image and displays all scale levels.
020 * To create a linear Gaussian scale space (without decimation)
021 * set 'P' = 1 and 'topLevel' to an arbitrary value.
022 * Consult the book for additional details.
023 */
024
025public class Show_GaussianScaleSpace implements PlugInFilter {
026        
027        static double sigma_0 = 1.6;    // initial scale level
028        static double sigma_s = 0.5;    // original image (sampling) scale
029        
030        static int K = 3;       // scale steps per octave
031        static int P = 4;       // number of octaves
032        static int botLevel = 0;        // index q of bottom level in each octave
033        static int topLevel = K;        // index q of top level in each octave
034
035        public int setup(String arg0, ImagePlus arg1) {
036                return DOES_8G + DOES_32 + NO_CHANGES;
037        }
038        
039        public void run(ImageProcessor ip) {
040                FloatProcessor fp = (FloatProcessor) ip.convertToFloat();
041                GaussianScaleSpace gss =
042                        new GaussianScaleSpace(fp, sigma_s, sigma_0, P, K, botLevel, topLevel);
043                gss.show("Gauss");
044        }
045
046}
