027/**
028 * This plugin demonstrates the use of the trigonometric method for calculating Fourier
029 * descriptors. The input image is assumed to contain a ROI whose (polygon) vertices are used 
030 * to calculate the Fourier descriptor, The actual image content is ignored, thus the 
031 * image type is irrelevant.
032 * A partial reconstruction of the contour (with 100 sample points) is calculated 
033 * from the Fourier descriptor 'fd'. Finally, a pair of invariant descriptors (contained in the 
034 * array 'fdAB') is calculated.
035 * 
036 * @author W. Burger
037 * @version 2013/05/30
038 */
039
040public class Fourier_Descriptor_IrregularSampling extends Fourier_Descriptor_RegularSampling { //implements PlugInFilter {
041                        
042        private ImagePlus img;
043
044        public int setup(String arg, ImagePlus img) { 
045        this.img = img;
046                return DOES_8G + ROI_REQUIRED + NO_CHANGES;
047        }
048        
049        public void run(ImageProcessor ip) {
050                if (!setParameters()) 
051                        return;
052                
053                Roi anyRoi = img.getRoi();
054                
055                // create the Fourier descriptor for 'anyRoi' with Mp coefficient pairs:
056                int Mp = FourierDescriptorPairs;
057                FourierDescriptor fd = new FourierDescriptorFromPolygon(anyRoi, Mp);
058
059                // reconstruct the corresponding shape with 100 contour points:
060                Complex[] R = fd.getReconstruction(ShapeReconstructionPoints);
061                
062                // create a pair of invariant descriptors (G^A, G^B):
063                FourierDescriptor[] fdAB = fd.makeInvariant();
064                FourierDescriptor fdA = fdAB[0];        // = G^A
065                FourierDescriptor fdB = fdAB[1];        // = G^B
066
067                // ----------------------------------------------------------------
068                // show various reconstructions (as a vector overlay)
069                // ----------------------------------------------------------------
070                
071                Overlay oly = new Overlay();
072                
073                if (DrawOriginalContour) {
074                        Roi roi = makeClosedPathShape(fd.getSamples(), 0.5, 0.5);
075                        roi.setStrokeColor(ContourColor);
076                        roi.setStrokeWidth(ContourStrokeWidth);
077                        oly.add(roi);
078                }
079                
080                if (DrawOriginalSamplePoints) {
081                        for (Complex c : fd.getSamples()) {
082                                ShapeRoi roi = makeCircleShape(c, SampleRadius, 0.5, 0.5);
083                                roi.setStrokeColor(SampleStrokeColor);
084                                roi.setStrokeWidth(SampleStrokeWidth);
085                                if (SampleFillColor != null) {
086                                        roi.setFillColor(SampleFillColor);
087                                }
088                                oly.add(roi);
089                        }
090                }
091                
092                if (DrawShapeCenter) {
093                        ShapeRoi roi = makeCrossShape(fd.getCoefficient(0), 2, 0.5, 0.5);
094                        roi.setStrokeColor(ReconstructionColor);
095                        roi.setStrokeWidth(ReconstructionStrokeWidth);
096                        oly.add(roi);
097                }
098        
099                if (DrawReconstruction) {
100                        ShapeRoi roi = makeClosedPathShape(R, 0.5, 0.5);
101                        roi.setStrokeColor(ReconstructionColor);
102                        roi.setStrokeWidth(ReconstructionStrokeWidth);
103                        oly.add(roi);
104                }
105                
106                if (DrawNormalizedShapes) {
107                        ShapeRoi roiA = makeClosedPathShape(fdA.getReconstruction(ShapeReconstructionPoints), 0.5, 0.5);
108                        roiA.setStrokeColor(CanonicalShapeAColor);
109                        roiA.setStrokeWidth(ReconstructionStrokeWidth);
110                        oly.add(roiA);
111                        ShapeRoi roiB = makeClosedPathShape(fdB.getReconstruction(ShapeReconstructionPoints), 0.5, 0.5);
112                        roiB.setStrokeColor(CanonicalShapeBColor);
113                        roiB.setStrokeWidth(ReconstructionStrokeWidth);
114                        oly.add(roiB);
115                }
116                
117                String title = img.getShortTitle() + "-Fourier-Descriptors";
118                ByteProcessor ip2 = (ByteProcessor) ip.duplicate();
119                ImagePlus im = new ImagePlus(title, ip2);       
120                if (ip2.isInvertedLut()) {
121                        ip2.invert();
122                        ip2.invertLut();
123                }
124                brighten(ip2, 220);
125                im.setOverlay(oly);
126                im.show();
127        }
128        
129        // helper methods ------------------------------------------------------
130        
131        Point2D[] getRoiPoints(Roi roi) {
132                Polygon poly = roi.getPolygon();
133                int[] xp = poly.xpoints;
134                int[] yp = poly.ypoints;
135                // copy vertices for all non-zero-length polygon segments:
136                List<Point> points = new ArrayList<Point>(xp.length);
137                points.add(new Point(xp[0], yp[0]));
138                int last = 0;
139                for (int i = 1; i < xp.length; i++) {
140                        if (xp[last] != xp[i] || yp[last] != yp[i]) {
141                                points.add(new Point(xp[i], yp[i]));
142                                last = i;
143                        }
144                }
145                // remove last point if the closing segment has zero length:
146                if (xp[last] == xp[0] && yp[last] == yp[0]) {
147                        points.remove(last);
148                }
149                return points.toArray(new Point2D[0]);
150        }
