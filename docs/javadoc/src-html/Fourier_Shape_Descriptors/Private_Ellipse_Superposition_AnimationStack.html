091        public void run(ImageProcessor ip1) {
092                int W = ip1.getWidth();
093                int H = ip1.getHeight();
094                
095                double imgScale = Math.sqrt(W*W + H*H) / Math.sqrt(200*200);    // relate image to "standard" size
096                ContourStrokeWidth = ContourStrokeWidth * imgScale;
097                SampleStrokeWidth = SampleStrokeWidth * imgScale;
098                ReconstructionStrokeWidth = ReconstructionStrokeWidth * imgScale;
099                ReconstructionRadius = ReconstructionRadius * imgScale;
100                
101                GenericDialog dlg = new GenericDialog("Fourier Shape Composition Parameters");
102                dlg.addMessage("Fourier decomosition parameters");
103                dlg.addNumericField("Number contour samples", NumberOfContourSamples, 0);
104                dlg.addNumericField("Number of FD pairs (min. 1)", NumberOfFourierDescriptorPairs, 0);
105                dlg.addMessage("Animation parameters");
106                dlg.addNumericField("Angular steps per revolution", StepsPerFullRevolution, 0);
107                dlg.addNumericField("Steps per second", StepsPerSecond, 0);
108                dlg.addMessage("Drawing parameters");
109                dlg.addNumericField("Reconstruction point radius", ReconstructionRadius, 1);
110                dlg.addCheckbox("Anti-alias graphics", AntialiasGraphics);
111                dlg.addMessage("Press ESC or close window to stop animation.");
112                dlg.showDialog();
113                if(dlg.wasCanceled()) return;
114                NumberOfContourSamples = (int) dlg.getNextNumber();
115                NumberOfFourierDescriptorPairs = Math.max(1, (int) dlg.getNextNumber());
116                StepsPerFullRevolution = (int) dlg.getNextNumber();
117                StepsPerSecond = (int) dlg.getNextNumber();
118                ReconstructionRadius = dlg.getNextNumber();
119                AntialiasGraphics = dlg.getNextBoolean();
120                dlg.addMessage("Press ESC or close window to stop animation.");
121
122                // ------------------------------------------------------------------------
123
124                ByteProcessor ip = (ByteProcessor) ip1.duplicate();
125
126                //  label regions and trace contours
127                RegionContourLabeling tracer = new RegionContourLabeling((ByteProcessor)ip);
128                //IJ.log("tracing done");
129
130                // extract contours and regions
131                List<Contour> outerContours = tracer.getOuterContours(true);
132                if (outerContours.isEmpty()) {
133                        IJ.error("no regions found");
134                        return;
135                }
136
137                Contour contr = outerContours.get(0);   // contour of largest region
138                Point2D[] polygon = new PolygonSampler().samplePolygonUniformly(contr.getPointArray(), NumberOfContourSamples);
139                FourierDescriptor fd = new FourierDescriptorUniform(polygon);
140                //fd.print();
141
142                String title = origTitle + String.format("-coeff%03d",NumberOfFourierDescriptorPairs);
143                ImagePlus im = new ImagePlus(title, ip);
144                if (ip.isInvertedLut()) {
145                        ip.invert();
146                        ip.invertLut();
147                }
148                brighten(ip, 220);
149                im.show();
150
151                long frameTimeMs = Math.round(1000.0/StepsPerSecond);
152
153                ImageStack stack = new ImageStack(ip.getWidth(), ip.getHeight());
154
155                for (int k = 0; k < StepsPerFullRevolution; k++) {
156
157                        if (IJ.escapePressed() || im.getWindow() == null || im.getWindow().isClosed()) {
158                                break;
159                        }
160
161                        double t = (double) k / StepsPerFullRevolution;
162                        long startTime = System.currentTimeMillis();
163                        Overlay oly = new Overlay();
164                        
165                        String timeString = String.format(Locale.US, "t=%.4f", t);
166
167                        if (ShowPathPosition) {
168                                TextRoi troi = new TextRoi(5, 0, timeString, font);
169                                troi.setStrokeColor(Color.black);
170                                oly.add(troi);
171                        }
172
173                        if (ShowOriginalContour) { // draw the original contour ------------------------------------
174                                oly.add(makeCentroidShape(contr));
175                        }
176
177                        if (ShowContourSamples) { // draw the contour sample points ------------------------------
178                                drawSamples(oly, fd.getSamples(), 0.5, 0.5);
179                        }
180
181                        if (MarkContourCentroid) { // mark the center ---------------------------------------------
182                                oly.add(makeCentroidShape(fd));
183                        }
184
185                        // --------------------------------------------------------------
186
187                        { // draw the reconstructed shape from FD-pairs 0,...,mMax -------
188                                int mMax = NumberOfFourierDescriptorPairs;
189                                Path2D path = fd.makeFourierPairsReconstruction(mMax);
190                                path.transform(CtrShift);
191                                ShapeRoi roi = new ShapeRoi(path);
192                                roi.setStrokeColor(ReconstructionColor);
193                                roi.setStrokeWidth(ReconstructionStrokeWidth);
194                                oly.add(roi);
195                        }
196
197                        //  draw superposition of ellipses ----------------------
198
199                        Complex cc = fd.getCoefficient(0);      // current ellipse center
200                        int mMax = NumberOfFourierDescriptorPairs;
201                        //                      RandomColorGenerator rcg = new RandomColorGenerator(1);
202
203                        CssColorSequencer csq = new CssColorSequencer();
204                        for (int m = 1; m <= mMax; m++) {
205                                //IJ.log("drawing ellipse " + m);
206                                //                              Color color = rcg.nextColor(); //makeRandomColor();
207                                Color color = csq.nextColor();
208                                Complex c1 = fd.getCoefficient(-m);
209                                Complex c2 = fd.getCoefficient(m);
210                                Path2D path = fd.makeEllipse(c1, c2, m, cc.re + 0.5, cc.im + 0.5);
211                                ShapeRoi rpoly = new ShapeRoi(path);
212                                rpoly.setStrokeColor(color);
213                                rpoly.setStrokeWidth(ReconstructionStrokeWidth/2);
214                                oly.add(rpoly);
215
216                                // mark the path position for t
217                                Complex c12 = fd.getEllipsePoint(c1, c2, m, t);
218                                //IJ.log("re: " + c12.re + " im: " + c12.im);
219                                double rad = ReconstructionRadius;
220                                Ellipse2D oval = 
221                                                new Ellipse2D.Double(c12.re + cc.re - rad + 0.5, c12.im + cc.im - rad + 0.5, 2 * rad, 2 * rad);
222                                Roi roi = new ShapeRoi(oval);
223                                roi.setFillColor(color);
224                                //roi.setStrokeWidth(ReconstructionStrokeWidth/2);
225                                oly.add(roi);
226                                cc = cc.add(c12);
227                        }
228
229                        im.setOverlay(oly);
230                        im.setAntialiasRendering(AntialiasGraphics);
231                        //im.updateAndDraw();
232                        
233                        IJ.wait(10);
234                        ImagePlus imFlat = im.flatten();
235                        IJ.wait(10);
236                        
237                        stack.addSlice(timeString, imFlat.getProcessor());
238
239                        long elapsedTime = System.currentTimeMillis() - startTime;
240                        int remainingTime = (int) (frameTimeMs - elapsedTime);
241
242                        if (remainingTime > 0) {
243                                //IJ.log("waiting " + remainingTime);
244                                //IJ.wait(remainingTime);
245                        }
246                }
247                
248                (new ImagePlus(title, stack)).show();
249
250        }

254        void drawSamples(Overlay oly, Complex[] samples, double dx, double dy) {
255                double rad = SampleRadius;
256                for (Complex c : samples) {
257                        Ellipse2D oval = new Ellipse2D.Double(c.re + dx - rad, c.im + dy - rad, 2 * rad, 2 * rad);
258                        Roi roi = new ShapeRoi(oval);
259                        roi.setFillColor(SampleColor);
260                        roi.setStrokeColor(SampleColor);
261                        roi.setStrokeWidth(SampleStrokeWidth);
262                        oly.add(roi);
263                }
264        }

267                for (BinaryRegion r: regions) {
268                        IJ.log("" + r);
269                }
270        }

272        void brighten(ByteProcessor ip, int minGray) {
273                float scale = (255 - minGray) / 255f;
274                int[] table = new int[256];
275                for (int i=0; i<256; i++) {
276                        table[i] = (int) Math.round(minGray + scale * i);
277
278                }
279                ip.applyTable(table);
280        }
281
282        void darken(ByteProcessor ip, int maxGray) {
283                float scale = maxGray / 255f;
284                int[] table = new int[256];
285                for (int i=0; i<256; i++) {
286                        table[i] = (int) Math.round(scale * i);
287
288                }
289                ip.applyTable(table);
290        }

294        ShapeRoi makeCentroidShape(FourierDescriptor fd) { 
295                int crossSize = 2;
296                Complex G0 = fd.getCoefficient(0);
297                double xc = G0.re;
298                double yc = G0.im;
299                Path2D path = new Path2D.Double();
300                path.moveTo(xc-crossSize, yc);
301                path.lineTo(xc+crossSize, yc);
302                path.moveTo(xc, yc-crossSize);
303                path.lineTo(xc, yc+crossSize);
304                path.transform(CtrShift);
305                ShapeRoi roi = new ShapeRoi(path);
306                roi.setStrokeColor(ReconstructionColor);
307                roi.setStrokeWidth(ReconstructionStrokeWidth);
308                return roi;
309        }
310
311        ShapeRoi makeCentroidShape(Contour contr) { // draw the original contour ------------------------------------
312                Path2D path = contr.getPolygonPath();
313                path.transform(CtrShift);
314                ShapeRoi roi = new ShapeRoi(path);
315                roi.setStrokeColor(ContourColor);
316                roi.setStrokeWidth(ContourStrokeWidth);
317                return roi;
318        }
