001package _Point_Fitting;
002
003import java.util.Random;
004
005import ij.IJ;
006import ij.ImagePlus;
007import ij.ImageStack;
008import ij.gui.GenericDialog;
009import ij.plugin.PlugIn;
010import ij.process.ByteProcessor;
011import ij.process.ImageProcessor;
012import imagingbook.lib.math.Matrix;
013import imagingbook.pub.geometry.basic.Point;
014import imagingbook.pub.geometry.fitting.AffineFit2D;
015import imagingbook.pub.geometry.fitting.LinearFit2D;
016import imagingbook.pub.geometry.fitting.ProcrustesFit;
017
018/**
019 * This IJ plugin creates a set of K random points, which are
020 * displayed in the first frame of a 2-image stack.
021 * The point set is transformed by a (fixed) affine transformation,
022 * the result is shown in frame 2.
023 * All point coordinates are integers, the transformed coordinates are rounded
024 * to their closest values.
025 * Then a linear fit is found between the two point sets, using either
026 * (a) an affine fitter or
027 * (b) a Procrustes fitter.
028 * The correspondence between the point sets is known.
029 * 
030 * @author WB
031 *
032 */
033public class Affine_Transform_Fitting_Demo implements PlugIn {
034        
035        static int W = 400;
036        static int H = 400;
037        static int K = 200;             // number of points
038        
039        static boolean allowTranslation = true;
040        static boolean allowScaling = true;
041        static boolean forceRotation = true;
042        
043        static final String title = Affine_Transform_Fitting_Demo.class.getSimpleName();
044        
045        private static final String[] FittingMethods = {"Affine Least Squares", "Procrustes"};
046        private static int theMethod = 0;
047        
048        
049//      static double[][] A =   // affine 2D transformation
050//              {{ 0.013,  1.088,  18.688 },
051//               {-1.000, -0.050, 127.500 }};
052        
053        static double a = 10 * 2 * Math.PI / 360;
054        static double ca = Math.cos(a);
055        static double sa = Math.sin(a);
056        static double S = 1.05;
057        
058        static double[][] A =   // affine 2D transformation
059                {{ S * ca, S * -sa, 18.688 },
060                 { S * sa, S *  ca, -7.123 }};
061        
062//      static double[][] A =   // affine 2D transformation
063//      {{ 1, 0, 0 },
064//       { 0, 1, 0 }};
065
066        public void run(String arg) {
067                if (!runDialog())
068                        return;
069                        
070                IJ.log("Fitter used: " + FittingMethods[theMethod]);
071                IJ.log("A (original) = \n" + Matrix.toString(A));
072                
073                // create the image stack
074                ImageStack stack = new ImageStack(W, H);
075                ImageProcessor ipP = new ByteProcessor(W, H);
076                ImageProcessor ipQ = new ByteProcessor(W, H);
077                stack.addSlice(ipP);
078                stack.addSlice(ipQ);
079                
080                // create K random points and draw them in ipP:
081                Random rg = new Random();
082                Point[] P = new Point[K];
083                for (int i = 0; i < K; i++) {
084                        int u = W/4 + rg.nextInt(W/2);
085                        int v = H/4 + rg.nextInt(H/2);
086                        P[i] = Point.create(u, v);
087                        ipP.putPixel(u, v, 255);
088                }
089                
090                // transform points: TODO: use an affine mapping instead!
091                Point[] Q = new Point[K];
092                for (int i = 0; i < K; i++) {
093                        double[] q = Matrix.multiply(A, Matrix.toHomogeneous(Q[i].toArray()));
094                        // we round coordinates to simulate a discrete grid transformation:
095                        int u = (int) Math.round(q[0]);
096                        int v = (int) Math.round(q[1]);
097                        Q[i] = Point.create(u, v);
098                        ipQ.putPixel(u, v, 255);
099                }
100                
101                new ImagePlus("Stack", stack).show();
102                
103                // select a fitter:
104                LinearFit2D fitter = null;
105                switch (theMethod) {
106                case 0: fitter = new AffineFit2D(P, Q); 
107                                break;
108                case 1: fitter = new ProcrustesFit(P, Q, allowTranslation, allowScaling, forceRotation); 
109                                break;
110                }
111                
112                double[][] A = fitter.getTransformationMatrix().getData();
113                IJ.log("A (estimated) = \n" + Matrix.toString(A));
114                IJ.log(String.format("e = %.6f", fitter.getError()));
115        }
116        
117        // ---------------------------------------------------------------------------
118
119        private boolean runDialog() {
120                GenericDialog gd = new GenericDialog(title);
121                gd.addChoice("Select fitting method", 
122                                FittingMethods, FittingMethods[theMethod]);
123                gd.showDialog(); 
124                if (gd.wasCanceled()) 
125                        return false;
126                theMethod = gd.getNextChoiceIndex();
127                return true;
128        }
129
130}
