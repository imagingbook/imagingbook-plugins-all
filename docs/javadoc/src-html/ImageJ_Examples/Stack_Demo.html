009package ImageJ_Examples;
010
011import ij.ImagePlus;
012import ij.ImageStack;
013import ij.plugin.filter.PlugInFilter;
014import ij.process.ImageProcessor;
015
016/**
017 * This ImageJ plugin demonstrates how to iterate over the frames (slices)
018 * of an image stack. In this case each frame is simply inverted. No new
019 * frames are added to the stack. The result is automatically re-displayed
020 * (unless {@literal NO_CHANGES} is specified).
021 * @author W. Burger
022 *
023 */
024public class Stack_Demo implements PlugInFilter {
025        
026        ImagePlus im = null;    // keep a reference to the associated ImagePlus object
027        
028        public int setup(String args, ImagePlus im) {
029                this.im = im;
030                return DOES_8G + STACK_REQUIRED;
031        }
032
033        public void run(ImageProcessor ignored) {
034                ImageStack stack = im.getImageStack();
035                int K = stack.getSize();
036                for (int k = 1; k <= K; k++) {  // NOTE: slices are numbered from 1,...,K !!
037                        ImageProcessor ip = stack.getProcessor(k);
038                        ip.invert();
039                }
040        }
041}
