009package ImageJ_Examples;
010
011import ij.ImagePlus;
012import ij.plugin.filter.PlugInFilter;
013import ij.process.ImageProcessor;
014import java.awt.Rectangle;
015
016/**
017 * This ImageJ plugin shows how an image operation can be confined
018 * to the currently selected region of interest (ROI). Note that
019 * the ROI attached to an ImageProcessor object is a rectangle,
020 * which is used to limit the range of visited pixels. In addition,
021 * if the image has a mask attached, it is used to check if the
022 * current pixel is inside an ROI of arbitrary shape.
023 * Note that masks have their own coordinate system whose origin
024 * is generally not the same as in the associated image.
025 * 
026 * @author WB
027 * @version 2015/03/23
028 */
029public class Roi_Demo implements PlugInFilter {
030        boolean showMask = true;
031
032        public int setup(String arg, ImagePlus imp) {
033                return DOES_RGB + ROI_REQUIRED;
034        }
035
036        public void run(ImageProcessor ip) {
037                Rectangle roi = ip.getRoi();
038                ImageProcessor mask = ip.getMask();
039                boolean hasMask = (mask != null);
040                if (hasMask && showMask) {
041                        (new ImagePlus("The Mask", mask)).show();
042                }
043
044                // ROI corner coordinates: 
045                int rLeft = roi.x;
046                int rTop = roi.y;
047                int rRight = rLeft + roi.width;
048                int rBottom = rTop + roi.height;
049
050                // process all pixels inside the ROI
051                for (int v = rTop; v < rBottom; v++) {
052                        for (int u = rLeft; u < rRight; u++) {
053                                if (!hasMask || mask.getPixel(u - rLeft, v - rTop) > 0) {
054                                        int p = ip.getPixel(u, v);
055                                        ip.putPixel(u, v, ~p);
056                                }
057
058                        }
059                }
060        }
061}
