009package ImageJ_Examples;
010
011import ij.IJ;
012import ij.ImagePlus;
013import ij.plugin.filter.PlugInFilter;
014import ij.process.ImageProcessor;
015
016/**
017 * This plugin demonstrates how to run another ImageJ command (plugin)
018 * from our own PlugInFilter using the IJ.run() method.
019 * Note that PlugInFilter locks the current image until is is finished,
020 * thus we need to unlock() it before running the other command.
021 * @author WB
022 */
023public class Run_Command_From_PlugInFilter implements PlugInFilter {
024        
025        ImagePlus im;
026
027        public int setup(String args, ImagePlus im) {
028                this.im = im;
029                return DOES_ALL;
030        }
031        
032        public void run(ImageProcessor ip) {
033                im.unlock();                                    // unlock the image to run other commands
034                IJ.run(im, "Invert", "");               // run another command
035                im.lock();                                              // lock the image again (just to be sure)
036                // ... continue with this plugin
037        }
038
039
040}
