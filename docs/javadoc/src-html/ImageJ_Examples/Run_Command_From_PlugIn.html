009package ImageJ_Examples;
010
011import ij.IJ;
012import ij.ImagePlus;
013import ij.plugin.PlugIn;
014
015/**
016 * This plugin demonstrates how to run another ImageJ command (plugin)
017 * from our own PlugIn using the IJ.run() method.
018 * @author WB
019 */
020public class Run_Command_From_PlugIn implements PlugIn {
021        
022    public void run(String args) {
023        ImagePlus im = IJ.getImage();
024        IJ.run(im, "Invert", "");       // run the "Invert" command on im
025        // IJ.run("Invert");            // alternatively, run "Invert" on the current image
026    }
027}
