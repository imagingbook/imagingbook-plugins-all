009package Geometric_Operations;
010
011import ij.ImagePlus;
012import ij.plugin.PlugIn;
013import ij.process.ByteProcessor;
014import imagingbook.lib.image.ImageGraphics;
015
016/**
017 * This ImageJ plugin draws a test grid in a newly created image.
018 * It uses anti-aliased drawing operations provided by 
019 * imagingbook's {@link ImageGraphics} class.
020 * 
021 * @author W. Burger
022 * @version 2020-01-08
023 */
024public class Draw_Test_Grid implements PlugIn {
025        
026        static int w = 400;
027        static int h = 400;
028        static int xStep = 20;
029        static int yStep = 20;
030        static int xStart = 100;
031        static int yStart = 100;
032        static int xN = 10;
033        static int yN = 10;
034        
035        static int foreground = 0;
036        static int background = 255;
037        
038    public void run(String arg) {
039        ByteProcessor ip = new ByteProcessor(w, h);
040        ip.setValue(background);
041        ip.fill();
042        
043        try (ImageGraphics g = new ImageGraphics(ip)) {
044                        g.setColor(foreground);
045                        g.setLineWidth(1.0);
046                        
047                        int y = yStart;
048                int x1 = xStart;
049                int x2 = xStart + xN * xStep;
050                        for (int j = 0; j <= yN; j++) {
051                                g.drawLine(x1, y, x2, y);
052                                y = y + yStep;
053                        }
054                        
055                        int x = xStart;
056                        int y1 = yStart;
057                        int y2 = yStart + yN * yStep;
058                        for (int i = 0; i <= xN; i++) {
059                                g.drawLine(x, y1, x, y2);
060                                x = x + xStep;
061                        }
062                        
063                        g.drawLine(0, 0, w - 1, h - 1);
064                        g.drawOval(xStart, yStart, w/2, h/2);
065        }
066        
067        new ImagePlus("Grid",ip).show();
068    }
069}
