009package Geometric_Operations;
010
011import ij.ImagePlus;
012import ij.plugin.filter.PlugInFilter;
013import ij.process.ImageProcessor;
014import imagingbook.lib.interpolation.InterpolationMethod;
015import imagingbook.pub.geometry.mappings.nonlinear.TwirlMapping;
016
017public class Transform_Twirl implements PlugInFilter {
018        static double angle = 0.75; // radians
019
020    public int setup(String arg, ImagePlus imp) {
021        return DOES_ALL;
022    }
023
024    public void run(ImageProcessor ip) {
025        int w = ip.getWidth();
026        int h = ip.getHeight();
027        double xcenter = 0.5 * w;
028                double ycenter = 0.5 * h;
029                double radius = Math.sqrt(xcenter*xcenter + ycenter*ycenter);
030                
031                TwirlMapping map = TwirlMapping.makeInverseMapping(xcenter, ycenter, angle, radius);
032                
033                map.applyTo(ip, InterpolationMethod.Bicubic);
034    }
035
036}
