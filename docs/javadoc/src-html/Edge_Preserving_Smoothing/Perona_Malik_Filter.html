009package Edge_Preserving_Smoothing;
010
011import ij.ImagePlus;
012import ij.gui.GenericDialog;
013import ij.plugin.filter.PlugInFilter;
014import ij.process.ImageProcessor;
015import imagingbook.lib.util.Enums;
016import imagingbook.pub.edgepreservingfilters.PeronaMalikFilter;
017import imagingbook.pub.edgepreservingfilters.PeronaMalikFilter.ColorMode;
018import imagingbook.pub.edgepreservingfilters.PeronaMalikFilter.Parameters;
019
020/**
021 * This plugin demonstrates the use of the PeronaMalikFilter class.
022 * This plugin works for all types of images and stacks.
023 * @author W. Burger
024 * @version 2013/05/30
025 */
026public class Perona_Malik_Filter implements PlugInFilter {
027
028        private Parameters params = new Parameters();
029
030        public int setup(String arg0, ImagePlus imp) {
031                if (!getParameters(imp))
032                        return DONE;
033                else
034                        return DOES_ALL + DOES_STACKS;
035        }
036
037        public void run(ImageProcessor ip) {
038                PeronaMalikFilter filter = new PeronaMalikFilter(params);
039                filter.applyTo(ip);
040        }
041        
042        private boolean getParameters(ImagePlus imp) {
043        boolean isColor = (imp.getType() == ImagePlus.COLOR_RGB);
044                GenericDialog gd = new GenericDialog("Anisotropic Diffusion Filter");
045                gd.addNumericField("Number of iterations", params.iterations, 0);
046                gd.addNumericField("Alpha (0,..,0.25)", params.alpha, 2);
047                gd.addNumericField("K", params.kappa, 0);
048                gd.addCheckbox("Smoother regions", params.smoothRegions);
049                if (isColor) {
050                        gd.addChoice("Color method", Enums.getEnumNames(ColorMode.class), params.colorMode.name());
051                        gd.addCheckbox("Use linear RGB", params.useLinearRgb);
052                }
053                gd.showDialog();
054                if (gd.wasCanceled())
055                        return false;
056                params.iterations = (int) Math.max(gd.getNextNumber(), 1);
057                params.alpha = (float) gd.getNextNumber();
058                params.kappa = (float) gd.getNextNumber();
059                params.smoothRegions = gd.getNextBoolean();
060                if (isColor) {
061                        params.colorMode = ColorMode.valueOf(gd.getNextChoice());
062                        params.useLinearRgb = gd.getNextBoolean();
063                }
064                return true;
065        }
066}
