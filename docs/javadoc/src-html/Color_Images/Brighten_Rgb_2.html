009package Color_Images;
010
011import ij.ImagePlus;
012import ij.plugin.filter.PlugInFilter;
013import ij.process.ColorProcessor;
014import ij.process.ImageProcessor;
015
016public class Brighten_Rgb_2 implements PlugInFilter {
017        static final int R = 0, G = 1, B = 2;   // component indices
018
019        public void run(ImageProcessor ip) {
020                //make sure image is of type ColorProcessor
021                ColorProcessor cp = (ColorProcessor) ip; 
022                int[] RGB = new int[3];
023
024                for (int v = 0; v < cp.getHeight(); v++) {
025                        for (int u = 0; u < cp.getWidth(); u++) {
026                                cp.getPixel(u, v, RGB); 
027                                RGB[R] = Math.min(RGB[R]+10, 255);  //add 10, limit to 255
028                                RGB[G] = Math.min(RGB[G]+10, 255);
029                                RGB[B] = Math.min(RGB[B]+10, 255);
030                                cp.putPixel(u, v, RGB); 
031                        }
032                }
033        }
034
035        public int setup(String arg, ImagePlus imp) {
036                return DOES_RGB;        // this plugin works on RGB images 
037        }
038}
