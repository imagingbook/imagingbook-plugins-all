009package Color_Images;
010
011import ij.ImagePlus;
012import ij.plugin.filter.PlugInFilter;
013import ij.process.ByteProcessor;
014import ij.process.ColorProcessor;
015import ij.process.ImageProcessor;
016
017/**
018 * This ImageJ plugin converts an RGB color image to a grayscale image
019 * using a specific set of component weights for caclulating the 
020 * luminance (luma) values.
021 * @author Wilbur
022 * @version 2013-12-02
023 */
024public class Convert_Rgb_To_Gray implements PlugInFilter {
025                
026        ImagePlus imp;
027        
028        public int setup(String arg, ImagePlus imp) {
029                this.imp = imp;
030                return DOES_RGB + NO_CHANGES;   
031        }
032
033        public void run(ImageProcessor ip) {
034                
035                ColorProcessor cp = (ColorProcessor) ip;
036                cp.setRGBWeights(0.2126, 0.7152, 0.0722); // ITU BR.709 luma weights
037                ByteProcessor bp = cp.convertToByteProcessor();
038                (new ImagePlus(imp.getShortTitle() + " (gray)", bp)).show();
039        }
040}
