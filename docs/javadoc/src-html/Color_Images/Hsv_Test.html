009package Color_Images;
010
011
012import ij.ImagePlus;
013import ij.plugin.filter.PlugInFilter;
014import ij.process.ColorProcessor;
015import ij.process.ImageProcessor;
016import imagingbook.pub.color.image.HsvConverter;
017
018/* This plugin rotates the color hue by 120 degrees */
019
020public class Hsv_Test implements PlugInFilter {
021
022        public int setup(String arg, ImagePlus imp) {
023                return DOES_RGB;
024        }
025
026        public void run(ImageProcessor ip) {
027                ColorProcessor cp = (ColorProcessor) ip;
028                final int[] RGB = new int[3];
029                
030                HsvConverter cc = new HsvConverter();
031                
032                for (int v = 0; v < cp.getHeight(); v++) {
033                        for (int u = 0; u < cp.getWidth(); u++) {
034                                cp.getPixel(u, v, RGB);
035
036                                float[] HSV = cc.RGBtoHSV (RGB);        // all HSV components are in [0,1]
037                                HSV[0] = (HSV[0] + 1.0f/3) % 1.0f;      // shift hue by 120 degrees
038                                int[] RGBnew = cc.HSVtoRGB(HSV);
039
040                                cp.putPixel(u, v, RGBnew);
041                        }
042                }
043        }
044}
