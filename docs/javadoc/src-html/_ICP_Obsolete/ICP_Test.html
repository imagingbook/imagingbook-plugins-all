096        private void clearPlot(ImageProcessor ip) {
097                //ImageProcessor ip = im.getProcessor();
098                ip.setColor(Color.white); 
099                ip.fill();
100        }
101
102        private void drawPoints(ImageProcessor ip, List<double[]> pnts, Color col) {
103                int w = 2;
104                ip.setColor(col);
105                ip.setLineWidth(2);
106                for (double[] p : pnts) {
107                        int u = (int) Math.round(p[0]);
108                        int v = (int) Math.round(p[1]);
109                        ip.drawLine(u-w, v, u+w, v);
110                        ip.drawLine(u, v-w, u, v+w);
111                }
112        }
113        
114        private void drawAssociations(ImageProcessor ip, List<double[]> pX, List<double[]> pY, int[] Assoc, Color col) {
115                ip.setColor(col);
116                ip.setLineWidth(1);
117                int i = 0;
118                for (double[] px : pX) {
119                        int j = Assoc[i];
120                        double[] py = pY.get(j);
121                        if (i == j) {
122                                drawLine(ip, px, py);
123                        }
124                        else {
125                                ip.setColor(Color.red);
126                                drawLine(ip, px, py);
127                                ip.setColor(col);
128                        }
129                        
130                        i = i + 1;
131                }
132        }
133        
134        private void drawLine(ImageProcessor ip, double[] a, double[] b) {
135                ip.drawLine(
136                        (int) Math.round(a[0]), (int) Math.round(a[1]),
137                        (int) Math.round(b[0]), (int) Math.round(b[1]));
138        }
139
140        // Rotation about the image center by angle theta + translation by (tx,ty)
141        private AffineMapping makeAffineTransformation() {
142                double ctr = 0.5 * size;
143                AffineMapping am = new AffineMapping();
144                am = am.concat(new Translation(-ctr, -ctr));
145                am = am.concat(new Rotation(theta));
146                am = am.concat(new Translation(ctr, ctr));
147                am = am.concat(new Translation(tx, ty));
148                return am;
149        }
150
151        private void  makeSamplePointsX() {
152                X = new ArrayList<double[]>(m);
153                for (int i = 0; i < m; i++) {
154                        double x = rg.nextInt(size);
155                        double y = rg.nextInt(size);
156                        X.add(new double[] {x, y});
157                }
158        }
159
160        private void makeSamplePointsY() {
161                Y = new ArrayList<double[]>(m);
162                for (double[] xi : X) {
163                        double[] xiT = A.applyTo(xi);
164                        xiT[0] = xiT[0] + sigma * rg.nextGaussian();
165                        xiT[1] = xiT[1] + sigma * rg.nextGaussian();
166                        Y.add(xiT);
167                }
168        }
169        
170        private List<double[]> mapPoints(RealMatrix M, List<double[]> X) {
171                List<double[]> XX = new ArrayList<double[]>(m);
172                for (double[] xi : X) {
173                        double[] xxi = M.operate(Matrix.toHomogeneous(xi));
174                        XX.add(xxi);
175                }
176                return XX;
177        }
