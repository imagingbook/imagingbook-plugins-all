009package Color_Quantization;
010
011import ij.ImagePlus;
012import ij.gui.GenericDialog;
013import ij.plugin.filter.PlugInFilter;
014import ij.process.ByteProcessor;
015import ij.process.ColorProcessor;
016import ij.process.ImageProcessor;
017import imagingbook.lib.ij.IjLogStream;
018import imagingbook.lib.util.Enums;
019import imagingbook.pub.color.quantize.ColorQuantizer;
020import imagingbook.pub.color.quantize.KMeansClusteringQuantizer;
021import imagingbook.pub.color.quantize.KMeansClusteringQuantizer.Parameters;
022import imagingbook.pub.color.quantize.KMeansClusteringQuantizer.SamplingMethod;
023
024
025/**
026 * ImageJ plugin demonstrating the use of the {@link KMeansClusteringQuantizer} class.
027 * 
028 * @author WB
029 * @version 2017/01/03
030 */
031public class KMeans_Quantization implements PlugInFilter {
032        
033        private static boolean CREATE_INDEXED_IMAGE = true; 
034        private static boolean CREATE_RGB_IMAGE = false;
035        private static boolean LIST_COLOR_TABLE = false;
036        
037        static {
038                IjLogStream.redirectSystem();
039        }
040        
041        public int setup(String arg, ImagePlus imp) {
042                return DOES_RGB + NO_CHANGES;
043        }
044        
045        public void run(ImageProcessor ip) {
046                Parameters params = new Parameters();
047        
048                if (!showDialog(params))
049                        return;
050                
051                ColorProcessor cp = (ColorProcessor) ip;
052                int[] pixels = (int[]) cp.getPixels();
053                
054                // create a quantizer object
055                ColorQuantizer quantizer = new KMeansClusteringQuantizer(pixels, params);
056                int nCols = quantizer.getColorMap().length;
057                
058                if (CREATE_INDEXED_IMAGE) {
059                        // quantize to an indexed color image
060                        ByteProcessor idxIp = quantizer.quantize(cp);
061                        (new ImagePlus("Quantized Index Color Image (" + nCols + " colors)", idxIp)).show();
062                }
063                
064                if (CREATE_RGB_IMAGE) {
065                        // quantize to a full-color RGB image
066                        int[] rgbPix = quantizer.quantize(pixels);
067                        ColorProcessor rgbIp = new ColorProcessor(cp.getWidth(), cp.getHeight(), rgbPix);
068                        (new ImagePlus("Quantized RGB Image (" + nCols + " colors)" , rgbIp)).show();
069                }
070                
071                if (LIST_COLOR_TABLE) {
072                        quantizer.listColorMap();
073                }
074        }
075        
076        private boolean showDialog(Parameters params) {
077                GenericDialog gd = new GenericDialog(KMeans_Quantization.class.getSimpleName());
078                gd.addNumericField("No. of colors (2,..,256)", params.maxColors, 0);
079                gd.addNumericField("Max. iterations", params.maxIterations, 0);
080                
081                String[] mNames = Enums.getEnumNames(SamplingMethod.class);
082                gd.addChoice("Sampling method", mNames, params.samplMethod.name());
083                
084                gd.addCheckbox("Create indexed color image", CREATE_INDEXED_IMAGE);
085                gd.addCheckbox("Create quantized RGB image", CREATE_RGB_IMAGE);
086                gd.addCheckbox("List quantized color table", LIST_COLOR_TABLE);
087                
088                gd.showDialog();
089                if (gd.wasCanceled())
090                        return false;
091                
092                int nc = (int) gd.getNextNumber();
093                nc = Math.min(nc, 255);
094                nc = Math.max(2, nc);
095
096                params.maxColors = nc;
097                params.maxIterations = (int) gd.getNextNumber();
098                params.samplMethod = SamplingMethod.valueOf(gd.getNextChoice());
099                
100                CREATE_INDEXED_IMAGE = gd.getNextBoolean();
101                CREATE_RGB_IMAGE = gd.getNextBoolean();
102                LIST_COLOR_TABLE = gd.getNextBoolean();
103                return true;
104        }
105}
