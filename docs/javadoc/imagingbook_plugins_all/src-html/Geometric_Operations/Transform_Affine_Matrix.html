011import ij.ImagePlus;
012import ij.gui.GenericDialog;
013import ij.plugin.filter.PlugInFilter;
014import ij.process.ImageProcessor;
015import imagingbook.lib.interpolation.InterpolationMethod;
016import imagingbook.pub.geometry.mappings.linear.AffineMapping;
017
018import java.awt.GridLayout;
019import java.awt.Label;
020import java.awt.Panel;
021import java.awt.TextField;
022
023/**
024 * ImageJ plugin for configurable affine image transformation.
025 * TODO: Polish the matrix dialogue input.
026 * 
027 * @author WB
028 * @version 2015/08/05
029 *
030 */
031public class Transform_Affine_Matrix implements PlugInFilter {
032        
033 
034    public int setup(String arg, ImagePlus imp) {
035        return DOES_ALL;
036    }
037
038    public void run(ImageProcessor ip) {
039        if (!showDialog())
040                return;
041        
042//        displayValues();
043        
044        double a11 = matrixValues[0][0];
045        double a12 = matrixValues[0][1];
046        double a13 = matrixValues[0][2];
047        double a21 = matrixValues[1][0];
048        double a22 = matrixValues[1][1];
049        double a23 = matrixValues[1][2];
050        
051        AffineMapping map = new AffineMapping(a11, a12, a13, a21, a22, a23, false);
052        map.applyTo(ip, InterpolationMethod.Bicubic);
053    }
054    
055    // Dialog example taken from http://rsbweb.nih.gov/ij/plugins/download/Dialog_Grid_Demo.java
056    
057    String[][] matrixNames = {
058                { "a11", "a12", "a13" },
059                { "a21", "a22", "a23" }};
060    
061    double[][] matrixValues = {
062                { 1, 0, 0 },
063                { 0, 1, 0 }};
064
065        
066        TextField[] tf = new TextField[matrixValues[0].length * matrixValues.length];
067//      double[] value = new double[gridSize];
068
069        private boolean showDialog() {
070                GenericDialog gd = new GenericDialog("Enter affine transformation matrix");
071                gd.addPanel(makePanel(gd));
072                gd.showDialog();
073                if (gd.wasCanceled())
074                        return false;
075                getValues();
076                return true;
077        }
078
079        private Panel makePanel(GenericDialog gd) {
080                int gridWidth = matrixValues[0].length;
081                int gridHeight = matrixValues.length;
082                Panel panel = new Panel();
083                panel.setLayout(new GridLayout(gridHeight, gridWidth * 2));
084                int i = 0;
085                for (int row = 0; row < gridHeight; row++) {
086                        for (int col = 0; col < gridWidth; col++) {
087                                tf[i] = new TextField("" + matrixValues[row][col]);
088                                panel.add(tf[i]);
089                                panel.add(new Label(matrixNames[row][col]));
090                                i++;
091                        }
092                }
093                return panel;
094        }
095        
096        private void getValues() {
097                int gridWidth = matrixValues[0].length;
098                int gridHeight = matrixValues.length;
099                int i = 0; 
100                for (int r = 0; r < gridHeight; r++) {
101                        for (int c = 0; c < gridWidth; c++) {
102                        String s = tf[i].getText();
103                        matrixValues[r][c] = getValue(s);
104                        i++;
105                        }
106                }
107        }
108        
109//      void displayValues() {
110//              int i = 0; 
111//              for (int r = 0; r < gridHeight; r++) {
112//                      for (int c = 0; c < gridWidth; c++) {
113//                              IJ.log(i + "  " + fieldValues[r][c]);
114//                              i++;
115//                      }
116//              }
117//      }
118
119        private double getValue(String valueAsText) {
120                Double d;
121                try {
122                        d = new Double(valueAsText);
123                } catch (NumberFormatException e) {
124                        d = null;
125                }
126                return (d == null) ? Double.NaN : d.doubleValue();
127        }
128
129}
