009package Pixel_Interpolation;
010
011import ij.ImagePlus;
012import ij.plugin.filter.PlugInFilter;
013import ij.process.ImageProcessor;
014import imagingbook.lib.image.ImageAccessor;
015import imagingbook.lib.image.OutOfBoundsStrategy;
016import static imagingbook.lib.image.OutOfBoundsStrategy.*;
017import imagingbook.lib.interpolation.InterpolationMethod;
018import static imagingbook.lib.interpolation.InterpolationMethod.*;
019
020
021public class Interpolator_Demo_Old implements PlugInFilter {
022        
023        static double dx = 0.5; // translation
024        static double dy = -3.5;
025        
026        static OutOfBoundsStrategy OBS = NearestBorder;
027        static InterpolationMethod IPM = BicubicSharp;
028 
029    public int setup(String arg, ImagePlus imp) {
030        return DOES_ALL + NO_CHANGES;
031    }
032
033    public void run(ImageProcessor source) {
034        final int w = source.getWidth();
035        final int h = source.getHeight();
036        
037        // create the target image (same type as source):
038        ImageProcessor target = source.createProcessor(w, h);
039        
040        // create an ImageAccessor for the source image:
041        ImageAccessor sA = ImageAccessor.create(source, OBS, IPM);
042        
043        // create an ImageAccessor for the target image:
044        ImageAccessor tA = ImageAccessor.create(target);
045        
046        // iterate over all pixels of the target image:
047        for (int u = 0; u < w; u++) {   // discrete target position (u,v)
048                for (int v = 0; v < h; v++) {
049                        double x = u + dx;      // continuous source position (x,y)
050                        double y = v + dy;
051                        float[] val = sA.getPix(x, y);
052                        tA.setPix(u, v, val);   // update target pixel
053                }
054        }
055        
056        // display the target image:
057        (new ImagePlus("Target", target)).show();
058    }
059}
